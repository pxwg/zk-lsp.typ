18/// Generate diagnostics for all @ID references in the document content.
19pub fn get_diagnostics(content: &str, index: &Arc<NoteIndex>, uri_path: &str) -> Vec<Diagnostic> {
20    let note_id = uri_path
21        .rsplit('/')
22        .next()
23        .and_then(|s| s.strip_suffix(".typ"))
24        .unwrap_or("");
25    let mut diagnostics = Vec::new();
26
27    for (line_num, line) in content.lines().enumerate() {
28        let refs = parser::find_all_refs(line);
29        for r in refs {
30            let range = Range {
31                start: Position {
32                    line: line_num as u32,
33                    character: parser::byte_to_utf16(line, r.start_char as usize),
34                },
35                end: Position {
36                    line: line_num as u32,
37                    character: parser::byte_to_utf16(line, r.end_char as usize),
38                },
39            };
40
41            let Some(info) = index.get(&r.id) else {
42                diagnostics.push(Diagnostic {
43                    range,
44                    severity: Some(DiagnosticSeverity::ERROR),
45                    source: Some("zk-lsp".into()),
46                    message: format!("Note @{} does not exist", r.id),
47                    ..Default::default()
48                });
49                continue;
50            };
51
52            if info.archived {
53                // Suppress if this note is a relation-target of the archived note
54                if info.relation_target.iter().any(|t| t == note_id) {
55                    continue;
56                }
57                let mut msg = format!("Note @{} is archived.", r.id);
58                if !info.relation_target.is_empty() {
59                    let targets = info
60                        .relation_target
61                        .iter()
62                        .map(|id| format!("@{id}"))
63                        .collect::<Vec<_>>()
64                        .join(", ");
65                    msg.push_str(&format!(" New ids: {targets}"));
66                }
67                let data = DiagnosticData {
68                    kind: "archived".into(),
69                    old_id: r.id.clone(),
70                    new_ids: Some(info.relation_target.clone()),
71                    replacement: None,
72                };
73                diagnostics.push(Diagnostic {
74                    range,
75                    severity: Some(DiagnosticSeverity::WARNING),
76                    source: Some("zk-lsp".into()),
77                    message: msg,
78                    data: Some(serde_json::to_value(data).unwrap()),
79                    ..Default::default()
80                });
81            } else if info.legacy {
82                // Suppress if this note is a relation-target of the legacy note
83                if info.relation_target.iter().any(|t| t == note_id) {
84                    continue;
85                }
86                // Suppress if the same line already mentions any successor.
87                let after = &line[r.end_char as usize..];
88                let has_successor_on_same_line = parser::find_all_refs(after)
89                    .into_iter()
90                    .any(|next| info.relation_target.iter().any(|id| id == &next.id));
91                let should_warn = !has_successor_on_same_line;
92
93                if should_warn {
94                    let mut msg = format!("Note @{} is legacy.", r.id);
95                    if !info.relation_target.is_empty() {
96                        let targets = info
97                            .relation_target
98                            .iter()
99                            .map(|id| format!("@{id}"))
100                            .collect::<Vec<_>>()
101                            .join(", ");
102                        msg.push_str(&format!(" New ids: {targets}"));
103                    }
104                    let data = DiagnosticData {
105                        kind: "legacy".into(),
106                        old_id: r.id.clone(),
107                        new_ids: Some(info.relation_target.clone()),
108                        replacement: None,
109                    };
110                    diagnostics.push(Diagnostic {
111                        range,
112                        severity: Some(DiagnosticSeverity::INFORMATION),
113                        source: Some("zk-lsp".into()),
114                        message: msg,
115                        data: Some(serde_json::to_value(data).unwrap()),
116                        ..Default::default()
117                    });
118                }
119            }
120        }
121    }
122
123    diagnostics
124}

126fn extract_toml_string_value(trimmed_line: &str) -> Option<&str> {
127    let eq_pos = trimmed_line.find('=')?;
128    let after_eq = trimmed_line[eq_pos + 1..].trim();
129    after_eq.strip_prefix('"')?.strip_suffix('"')
130}

132fn extract_toml_field_name(trimmed_line: &str) -> Option<&str> {
133    trimmed_line.split_once('=').map(|(field, _)| field.trim())
134}

136/// Validate TOML metadata block fields and produce diagnostics.
137pub fn get_schema_diagnostics(content: &str, index: &Arc<NoteIndex>) -> Vec<Diagnostic> {
138    let lines: Vec<&str> = content.lines().collect();
139    let Some(block) = parser::find_toml_metadata_block(content) else {
140        return vec![Diagnostic {
141            range: Range {
142                start: Position {
143                    line: 0,
144                    character: 0,
145                },
146                end: Position {
147                    line: 0,
148                    character: lines.first().map(|l| l.len()).unwrap_or(0) as u32,
149                },
150            },
151            severity: Some(DiagnosticSeverity::ERROR),
152            source: Some("zk-lsp".into()),
153            message: "Missing `zk-metadata` TOML block".to_string(),
154            ..Default::default()
155        }];
156    };
157
158    let title_line_idx = lines
159        .iter()
160        .enumerate()
161        .skip(block.end_line + 1)
162        .find_map(|(idx, line)| parser::RE_TITLE.is_match(line).then_some(idx));
163
164    if title_line_idx.is_none() {
165        return vec![Diagnostic {
166            range: Range {
167                start: Position {
168                    line: block.end_line as u32,
169                    character: 0,
170                },
171                end: Position {
172                    line: block.end_line as u32,
173                    character: 0,
174                },
175            },
176            severity: Some(DiagnosticSeverity::ERROR),
177            source: Some("zk-lsp".into()),
178            message: "Missing note title heading (`= Title <ID>`)".to_string(),
179            ..Default::default()
180        }];
181    }
182
183    // Try to parse as TOML; if invalid, return a single parse-error diagnostic
184    if let Err(e) = block.toml_content.parse::<toml::Value>() {
185        return vec![Diagnostic {
186            range: Range {
187                start: Position {
188                    line: block.start_line as u32,
189                    character: 0,
190                },
191                end: Position {
192                    line: block.end_line as u32,
193                    character: 0,
194                },
195            },
196            severity: Some(DiagnosticSeverity::ERROR),
197            source: Some("zk-lsp".into()),
198            message: format!("TOML parse error: {e}"),
199            ..Default::default()
200        }];
201    }
202
203    let mut diagnostics = Vec::new();
204    let toml_line_count = block.toml_content.lines().count();
205    let toml_start = block.end_line.saturating_sub(toml_line_count);
206    let expected_fields = [
207        ("schema-version", "  schema-version = 1\n"),
208        ("aliases", "  aliases = []\n"),
209        ("abstract", "  abstract = \"\"\n"),
210        ("keywords", "  keywords = []\n"),
211        ("generated", "  generated = true\n"),
212        ("checklist-status", "  checklist-status = \"none\"\n"),
213        ("relation", "  relation = \"active\"\n"),
214        ("relation-target", "  relation-target = []\n"),
215    ];
216    let mut present_fields = std::collections::HashMap::new();
217
218    // Per-line field validation
219    for (i, toml_line) in block.toml_content.lines().enumerate() {
220        let file_line = toml_start + i;
221        let file_line_text = lines.get(file_line).copied().unwrap_or("");
222        let trimmed = toml_line.trim_start();
223        if let Some(field_name) = extract_toml_field_name(trimmed) {
224            if let Some((field, _)) = expected_fields
225                .iter()
226                .find(|(field, _)| *field == field_name)
227            {
228                present_fields.insert(*field, file_line);
229            }
230        }
231
232        let line_range = Range {
233            start: Position {
234                line: file_line as u32,
235                character: 0,
236            },
237            end: Position {
238                line: file_line as u32,
239                character: file_line_text.len() as u32,
240            },
241        };
242
243        if trimmed.starts_with("checklist-status") {
244            if let Some(val) = extract_toml_string_value(trimmed) {
245                if crate::parser::ChecklistStatus::from_str(val).is_none() {
246                    diagnostics.push(Diagnostic {
247                        range: line_range,
248                        severity: Some(DiagnosticSeverity::ERROR),
249                        source: Some("zk-lsp".into()),
250                        message: format!(
251                            "Invalid checklist-status \"{val}\". Expected: none, todo, wip, done"
252                        ),
253                        ..Default::default()
254                    });
255                }
256            }
257        } else if trimmed.starts_with("relation") && !trimmed.starts_with("relation-target") {
258            if let Some(val) = extract_toml_string_value(trimmed) {
259                if !["active", "archived", "legacy"].contains(&val) {
260                    diagnostics.push(Diagnostic {
261                        range: line_range,
262                        severity: Some(DiagnosticSeverity::ERROR),
263                        source: Some("zk-lsp".into()),
264                        message: format!(
265                            "Invalid relation \"{val}\". Expected: active, archived, legacy"
266                        ),
267                        ..Default::default()
268                    });
269                }
270            }
271        }
272    }
273
274    for (idx, (field, replacement)) in expected_fields.iter().enumerate() {
275        if present_fields.contains_key(field) {
276            continue;
277        }
278        let insert_line = expected_fields[idx + 1..]
279            .iter()
280            .find_map(|(next_field, _)| present_fields.get(next_field).copied())
281            .unwrap_or(block.end_line);
282        diagnostics.push(Diagnostic {
283            range: Range {
284                start: Position {
285                    line: insert_line as u32,
286                    character: 0,
287                },
288                end: Position {
289                    line: insert_line as u32,
290                    character: 0,
291                },
292            },
293            severity: Some(DiagnosticSeverity::INFORMATION),
294            source: Some("zk-lsp".into()),
295            message: format!("Missing TOML field `{field}`"),
296            data: Some(
297                serde_json::to_value(DiagnosticData {
298                    kind: "missing-toml-field".into(),
299                    old_id: (*field).to_string(),
300                    new_ids: None,
301                    replacement: Some((*replacement).to_string()),
302                })
303                .unwrap(),
304            ),
305            ..Default::default()
306        });
307    }
308
309    // Semantic checks using parsed values
310    if let Some(parsed) = parser::parse_toml_metadata(&block.toml_content) {
311        use crate::parser::Relation;
312
313        // relation != "active" but relation-target is empty → WARNING on relation line
314        if parsed.relation != Relation::Active && parsed.relation_target.is_empty() {
315            if let Some((i, _)) = block.toml_content.lines().enumerate().find(|(_, l)| {
316                let t = l.trim_start();
317                t.starts_with("relation") && !t.starts_with("relation-target")
318            }) {
319                let file_line = toml_start + i;
320                let file_line_text = lines.get(file_line).copied().unwrap_or("");
321                diagnostics.push(Diagnostic {
322                    range: Range {
323                        start: Position {
324                            line: file_line as u32,
325                            character: 0,
326                        },
327                        end: Position {
328                            line: file_line as u32,
329                            character: file_line_text.len() as u32,
330                        },
331                    },
332                    severity: Some(DiagnosticSeverity::WARNING),
333                    source: Some("zk-lsp".into()),
334                    message: "relation is not 'active' but relation-target is empty".to_string(),
335                    ..Default::default()
336                });
337            }
338        }
339
340        // relation-target IDs not found in index → WARNING on relation-target line
341        let unknown_ids: Vec<&str> = parsed
342            .relation_target
343            .iter()
344            .filter(|id| !id.is_empty() && index.get(id).is_none())
345            .map(String::as_str)
346            .collect();
347
348        if !unknown_ids.is_empty() {
349            if let Some((i, _)) = block
350                .toml_content
351                .lines()
352                .enumerate()
353                .find(|(_, l)| l.trim_start().starts_with("relation-target"))
354            {
355                let file_line = toml_start + i;
356                let file_line_text = lines.get(file_line).copied().unwrap_or("");
357                for id in unknown_ids {
358                    diagnostics.push(Diagnostic {
359                        range: Range {
360                            start: Position {
361                                line: file_line as u32,
362                                character: 0,
363                            },
364                            end: Position {
365                                line: file_line as u32,
366                                character: file_line_text.len() as u32,
367                            },
368                        },
369                        severity: Some(DiagnosticSeverity::WARNING),
370                        source: Some("zk-lsp".into()),
371                        message: format!("Note @{id} does not exist in the index"),
372                        ..Default::default()
373                    });
374                }
375            }
376        }
377    }
378
379    diagnostics
380}

382/// Generate a HINT diagnostic for an orphan note.
383///
384/// A note is orphan only when BOTH conditions hold:
385/// 1. No other note references it (no backlinks in the index)
386/// 2. It has no outgoing `@ID` references itself
387///
388/// Returns `None` if the note is not in the index or is not fully isolated.
389pub fn get_orphan_diagnostic(
390    content: &str,
391    uri_path: &str,
392    index: &Arc<NoteIndex>,
393) -> Option<Diagnostic> {
394    let note_id = uri_path
395        .rsplit('/')
396        .next()
397        .and_then(|s| s.strip_suffix(".typ"))?;
398
399    // Only flag notes that are in the index
400    if index.get(note_id).is_none() {
401        return None;
402    }
403
404    // Not an orphan if it has inbound links
405    if !index.get_backlinks(note_id).is_empty() {
406        return None;
407    }
408
409    // Not an orphan if it has outgoing links
410    if !parser::find_all_refs_filtered(content).is_empty() {
411        return None;
412    }
413
414    // Find the title line (contains `<{note_id}>`)
415    let needle = format!("<{note_id}>");
416    let (line_num, _line_text) = content
417        .lines()
418        .enumerate()
419        .find(|(_, l)| l.contains(&needle))?;
420
421    Some(Diagnostic {
422        range: Range {
423            start: Position {
424                line: line_num as u32,
425                character: 0,
426            },
427            end: Position {
428                line: line_num as u32,
429                character: 0,
430            },
431        },
432        severity: Some(DiagnosticSeverity::HINT),
433        source: Some("zk-lsp".into()),
434        message: format!("Orphan note: no inbound or outbound @ID references"),
435        ..Default::default()
436    })
437}

439/// Generate WARNING diagnostics for Ref checklist items that are non-leaf nodes.
440///
441/// A RefItem (`- [ ] @ID`) must always be a leaf. If it has child items (next item
442/// with strictly greater indent), the @ID targets will be semantically ignored by
443/// the leaf rule, silently breaking the dependency.
444pub fn get_reconcile_diagnostics(
445    content: &str,
446    file_path: &std::path::Path,
447    reconcile_diagnostics: &[ReconcileDiagnostic],
448) -> Vec<Diagnostic> {
449    let mut diagnostics = Vec::new();
450
451    for diag in reconcile_diagnostics {
452        let Some(location) = &diag.location else {
453            continue;
454        };
455        if location.file_path != file_path {
456            continue;
457        }
458
459        let line_text = content.lines().nth(location.line).unwrap_or("");
460        diagnostics.push(Diagnostic {
461            range: Range {
462                start: Position {
463                    line: location.line as u32,
464                    character: parser::byte_to_utf16(line_text, location.byte_start as usize),
465                },
466                end: Position {
467                    line: location.line as u32,
468                    character: parser::byte_to_utf16(line_text, location.byte_end as usize),
469                },
470            },
471            severity: Some(match diag.severity {
472                ReconcileSeverity::Error => DiagnosticSeverity::ERROR,
473            }),
474            source: Some("zk-lsp".into()),
475            message: diag.message.clone(),
476            related_information: if diag.related_locations.is_empty() {
477                None
478            } else {
479                Some(
480                    diag.related_locations
481                        .iter()
482                        .map(|related| {
483                            let related_line = if related.file_path == file_path {
484                                content.lines().nth(related.line).unwrap_or("").to_string()
485                            } else {
486                                std::fs::read_to_string(&related.file_path)
487                                    .ok()
488                                    .and_then(|content| {
489                                        content.lines().nth(related.line).map(str::to_string)
490                                    })
491                                    .unwrap_or_default()
492                            };
493                            DiagnosticRelatedInformation {
494                                location: Location {
495                                    uri: Url::from_file_path(&related.file_path)
496                                        .expect("valid file path"),
497                                    range: Range {
498                                        start: Position {
499                                            line: related.line as u32,
500                                            character: parser::byte_to_utf16(
501                                                &related_line,
502                                                related.byte_start as usize,
503                                            ),
504                                        },
505                                        end: Position {
506                                            line: related.line as u32,
507                                            character: parser::byte_to_utf16(
508                                                &related_line,
509                                                related.byte_end as usize,
510                                            ),
511                                        },
512                                    },
513                                },
514                                message: "other dependency edge in the same cycle".to_string(),
515                            }
516                        })
517                        .collect(),
518                )
519            },
520            ..Default::default()
521        });
522    }
523
524    diagnostics
525}
