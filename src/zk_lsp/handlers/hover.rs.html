18fn get_metadata_hover_with_loader<F>(
19    content: &str,
20    position: Position,
21    index: &Arc<NoteIndex>,
22    load_note: F,
23) -> Option<Hover>
24where
25    F: Fn(&std::path::Path) -> Option<String>,
26{
27    let id_pos =
28        metadata::id_at_position(content, position.line as usize, position.character as usize)?;
29    let info = index.notes.get(&id_pos.target_id)?;
30    let note_content = load_note(&info.path)?;
31    let preview_content = extract_preview_body(&note_content);
32
33    let markdown = format!(
34        "**{}** `{}`\n\n```typst\n{}\n```",
35        info.title,
36        info.id,
37        preview_content.trim_end()
38    );
39
40    Some(Hover {
41        contents: HoverContents::Markup(MarkupContent {
42            kind: MarkupKind::Markdown,
43            value: markdown,
44        }),
45        range: None,
46    })
47}
48
49fn extract_preview_body(content: &str) -> String {
50    let Some(header) = parser::parse_header(content) else {
51        return content.to_string();
52    };
53    let lines: Vec<&str> = content.lines().collect();
54    lines[header.title_line_idx..].join("\n")
55}

65    fn make_index(id: &str, title: &str, path: PathBuf) -> Arc<NoteIndex> {
66        let idx = NoteIndex::new(Arc::new(tokio::sync::RwLock::new(WikiConfig::from_root(
67            PathBuf::from("/tmp"),
68        ))));
69        idx.notes.insert(
70            id.to_string(),
71            NoteInfo {
72                id: id.to_string(),
73                title: title.to_string(),
74                archived: false,
75                legacy: false,
76                alt_id: None,
77                evo_id: None,
78                relation_target: vec![],
79                aliases: vec![],
80                keywords: vec![],
81                abstract_text: None,
82                checklist_status: None,
83                path,
84            },
85        );

107        let path = PathBuf::from("/virtual/2603110001.typ");

113                character: 22,
