17/// Apply a list of byte-range edits to `content`.
18///
19/// Each edit is `(start_byte, end_byte, replacement_text)`.
20/// Edits must be non-overlapping. They are applied from last to first so byte
21/// offsets remain valid throughout.
22///
23/// Returns `Err` if any edit is invalid (out of bounds, inverted range, or overlap).
24pub fn apply_byte_edits(content: &str, edits: &[(usize, usize, String)]) -> anyhow::Result<String> {
25    let len = content.len();
26    for (start, end, _) in edits {
27        anyhow::ensure!(start <= end, "edit has start ({start}) > end ({end})");
28        anyhow::ensure!(*end <= len, "edit end ({end}) out of bounds (len={len})");
29    }
30    // Sort by start ascending, check no overlaps
31    let mut sorted: Vec<&(usize, usize, String)> = edits.iter().collect();
32    sorted.sort_by_key(|(s, _, _)| *s);
33    for w in sorted.windows(2) {
34        anyhow::ensure!(
35            w[0].1 <= w[1].0,
36            "edits overlap: [{}, {}) and [{}, {})",
37            w[0].0,
38            w[0].1,
39            w[1].0,
40            w[1].1
41        );
42    }
43    // Apply in reverse order so earlier byte offsets remain valid
44    let mut result = content.to_string();
45    for (start, end, text) in sorted.iter().rev() {
46        result.replace_range(start..end, text);
47    }
48    Ok(result)
49}

51/// Render a `toml::Value` as an inline TOML literal (no newlines).
52/// Used by `apply_metadata_patch` for targeted in-place key replacement.
53fn toml_value_inline(value: &toml::Value) -> anyhow::Result<String> {
54    match value {
55        toml::Value::String(s) => {
56            let escaped = s.replace('\\', "\\\\").replace('"', "\\\"");
57            Ok(format!("\"{escaped}\""))
58        }
59        toml::Value::Integer(n) => Ok(n.to_string()),
60        toml::Value::Float(f) => Ok(f.to_string()),
61        toml::Value::Boolean(b) => Ok(b.to_string()),
62        toml::Value::Array(arr) => {
63            let items: Vec<String> = arr
64                .iter()
65                .map(toml_value_inline)
66                .collect::<anyhow::Result<_>>()?;
67            Ok(format!("[{}]", items.join(", ")))
68        }
69        _ => anyhow::bail!("nested TOML tables are not supported in a metadata patch"),
70    }
71}

73/// Apply a metadata patch to a TOML-format note using targeted in-place replacement.
74///
75/// For each `(key, value)` in `patch`, finds the key's line within the TOML block
76/// and replaces only that line — preserving key order, indentation, and all other lines.
77///
78/// Handles two TOML representations of dotted keys (both are generated by `zk-lsp new`):
79/// - Dotted notation: `user.verified = "false"`
80/// - Section notation: `[user]` followed by `verified = "false"` on a later line
81///
82/// Returns `Err` if the note has no TOML metadata block or a key is not found in the block.
83pub fn apply_metadata_patch(
84    content: &str,
85    patch: &HashMap<String, toml::Value>,
86) -> anyhow::Result<String> {
87    let trailing_newline = content.ends_with('\n');
88    let mut result_lines: Vec<String> = content.lines().map(|l| l.to_string()).collect();
89
90    'patch: for (key, value) in patch {
91        let (block_start, block_end) = find_metadata_block_line_range(&result_lines)?;
92        let val_str = toml_value_inline(value)?;
93
94        // Pass 1: dotted-key notation — `user.verified = "false"`
95        for i in block_start..=block_end {
96            let line = result_lines
97                .get(i)
98                .map(String::as_str)
99                .ok_or_else(|| anyhow::anyhow!("metadata block line {i} missing"))?;
100            let trimmed = line.trim_start();
101            if trimmed.starts_with(&format!("{key} =")) || trimmed.starts_with(&format!("{key}=")) {
102                let indent_len = line.len() - trimmed.len();
103                result_lines[i] = format!("{}{key} = {val_str}", " ".repeat(indent_len));
104                continue 'patch;
105            }
106        }
107
108        // Pass 2: section-header notation — `[table]` … `field = value`
109        // Applies when key contains exactly one dot, e.g. "user.verified".
110        if let Some(dot_pos) = key.find('.') {
111            let table_name = &key[..dot_pos];
112            let field_name = &key[dot_pos + 1..];
113            // Only handle single-level nesting (no further dots in field_name).
114            if !field_name.contains('.') {
115                let section_header = format!("[{table_name}]");
116                let mut in_section = false;
117                for i in block_start..=block_end {
118                    let line = result_lines
119                        .get(i)
120                        .map(String::as_str)
121                        .ok_or_else(|| anyhow::anyhow!("metadata block line {i} missing"))?;
122                    let trimmed = line.trim_start();
123                    if trimmed == section_header {
124                        in_section = true;
125                        continue;
126                    }
127                    if in_section {
128                        // Another section header ends this one.
129                        if trimmed.starts_with('[') {
130                            break;
131                        }
132                        if trimmed.starts_with(&format!("{field_name} ="))
133                            || trimmed.starts_with(&format!("{field_name}="))
134                        {
135                            let indent_len = line.len() - trimmed.len();
136                            result_lines[i] =
137                                format!("{}{field_name} = {val_str}", " ".repeat(indent_len));
138                            continue 'patch;
139                        }
140                    }
141                }
142
143                insert_missing_section_key(
144                    &mut result_lines,
145                    block_start,
146                    block_end,
147                    table_name,
148                    field_name,
149                    &val_str,
150                )?;
151                continue 'patch;
152            }
153        }
154
155        insert_missing_top_level_key(&mut result_lines, block_start, block_end, key, &val_str)?;
156    }
157
158    let mut out = result_lines.join("\n");
159    if trailing_newline {
160        out.push('\n');
161    }
162    Ok(out)
163}

165fn find_metadata_block_line_range(lines: &[String]) -> anyhow::Result<(usize, usize)> {
166    let start = lines
167        .iter()
168        .position(|line| {
169            line.trim().starts_with("#let zk-metadata") && line.contains("toml(bytes(")
170        })
171        .ok_or_else(|| anyhow::anyhow!("no TOML metadata block found"))?;
172
173    let open_fence = lines
174        .iter()
175        .enumerate()
176        .skip(start + 1)
177        .find_map(|(idx, line)| (line.trim() == "```toml").then_some(idx))
178        .ok_or_else(|| anyhow::anyhow!("missing opening TOML code fence"))?;
179
180    let close_fence = lines
181        .iter()
182        .enumerate()
183        .skip(open_fence + 1)
184        .find_map(|(idx, line)| line.trim().starts_with("```").then_some(idx))
185        .ok_or_else(|| anyhow::anyhow!("missing closing TOML code fence"))?;
186
187    Ok((open_fence + 1, close_fence))
188}

190fn first_section_or_close(
191    lines: &[String],
192    block_start: usize,
193    block_end: usize,
194) -> anyhow::Result<usize> {
195    for i in block_start..=block_end {
196        let trimmed = lines
197            .get(i)
198            .map(String::as_str)
199            .ok_or_else(|| anyhow::anyhow!("metadata block line {i} missing"))?
200            .trim_start();
201        if trimmed.starts_with('[') || trimmed.starts_with("```") {
202            return Ok(i);
203        }
204    }
205    anyhow::bail!("malformed TOML metadata block")
206}

208fn insert_missing_top_level_key(
209    lines: &mut Vec<String>,
210    block_start: usize,
211    block_end: usize,
212    key: &str,
213    val_str: &str,
214) -> anyhow::Result<()> {
215    let mut insert_at = first_section_or_close(lines, block_start, block_end)?;
216    if insert_at > block_start && lines[insert_at - 1].trim().is_empty() {
217        insert_at -= 1;
218    }
219    lines.insert(insert_at, format!("  {key} = {val_str}"));
220    Ok(())
221}

223fn insert_missing_section_key(
224    lines: &mut Vec<String>,
225    block_start: usize,
226    block_end: usize,
227    table_name: &str,
228    field_name: &str,
229    val_str: &str,
230) -> anyhow::Result<()> {
231    let section_header = format!("[{table_name}]");
232    let mut section_line = None;
233    let mut insert_at = None;
234
235    for i in block_start..=block_end {
236        let trimmed = lines
237            .get(i)
238            .map(String::as_str)
239            .ok_or_else(|| anyhow::anyhow!("metadata block line {i} missing"))?
240            .trim_start();
241        if section_line.is_none() && trimmed == section_header {
242            section_line = Some(i);
243            continue;
244        }
245        if section_line.is_some() && (trimmed.starts_with('[') || trimmed.starts_with("```")) {
246            insert_at = Some(i);
247            break;
248        }
249    }
250
251    if let Some(section_line) = section_line {
252        let insert_at = insert_at.unwrap_or(block_end);
253        let _ = section_line;
254        lines.insert(insert_at, format!("  {field_name} = {val_str}"));
255        return Ok(());
256    }
257
258    let insert_at = first_section_or_close(lines, block_start, block_end)?;
259    if insert_at > 0 && !lines[insert_at - 1].trim().is_empty() {
260        lines.insert(insert_at, String::new());
261        lines.insert(insert_at + 1, format!("  [{table_name}]"));
262        lines.insert(insert_at + 2, format!("  {field_name} = {val_str}"));
263    } else {
264        lines.insert(insert_at, format!("  [{table_name}]"));
265        lines.insert(insert_at + 1, format!("  {field_name} = {val_str}"));
266    }
267    Ok(())
268}

270/// Format `content` by running hooks in sequence:
271/// 1. Built-in default hooks (checklist.lua + relation_status.lua, embedded at compile time),
272///    unless `config.zk_config.disable_default_hooks` is true.
273/// 2. User-configured file hooks from `config.zk_config.hooks`, loaded at runtime.
274///
275/// Cross-file ref-checkbox sync (`@ID` items) is intentionally NOT performed here;
276/// that is the exclusive responsibility of the `reconcile` command.
277///
278/// On any hook error the step is skipped and a warning is emitted; the original
279/// content (or the output of the previous step) is passed through unchanged.
280pub async fn format_content(content: &str, config: &WikiConfig) -> String {
281    let zk = &config.zk_config;
282    let mut current = content.to_string();
283    if !zk.disable_default_hooks {
284        current = run_default_hooks_with_metadata_fields(&current, &zk.metadata.fields);
285    }
286    current = run_hooks(&current, &zk.hooks, &zk.metadata.fields);
287    current
288}

290/// Run the built-in embedded hooks (checklist.lua + relation_status.lua).
291#[allow(dead_code)]
292pub(crate) fn run_default_hooks(content: &str) -> String {
293    run_default_hooks_with_metadata_fields(content, &[])
294}
295
296pub(crate) fn run_default_hooks_with_metadata_fields(
297    content: &str,
298    metadata_fields: &[MetadataFieldConfig],
299) -> String {
300    let hooks: &[(&str, &str)] = &[
301        ("checklist", DEFAULT_CHECKLIST_HOOK),
302        ("relation_status", DEFAULT_RELATION_HOOK),
303    ];
304    let mut current = content.to_string();
305    for (name, src) in hooks {
306        let runner = match HookRunner::load_str(src) {
307            Ok(r) => r,
308            Err(e) => {
309                tracing::warn!("default hook '{name}' load error: {e}");
310                continue;
311            }
312        };
313        let input = build_hook_note_input_with_metadata_fields(&current, metadata_fields);
314        let result = match runner.run(&input) {
315            Ok(r) => r,
316            Err(e) => {
317                tracing::warn!("default hook '{name}' run error: {e}");
318                continue;
319            }
320        };
321        match apply_hook_result(&result, &current) {
322            Ok(out) => current = out,
323            Err(e) => tracing::warn!("default hook '{name}' apply error: {e}"),
324        }
325    }
326    current
327}

329/// Run user-configured file hooks loaded at runtime. No-op if `hook_paths` is empty.
330pub(crate) fn run_hooks(
331    content: &str,
332    hook_paths: &[PathBuf],
333    metadata_fields: &[MetadataFieldConfig],
334) -> String {
335    let mut current = content.to_string();
336    for path in hook_paths {
337        let name = path
338            .file_name()
339            .map(|n| n.to_string_lossy())
340            .unwrap_or_default();
341        let runner = match HookRunner::load_file(path) {
342            Ok(r) => r,
343            Err(e) => {
344                tracing::warn!("hook '{name}' load error: {e}");
345                continue;
346            }
347        };
348        let input = build_hook_note_input_with_metadata_fields(&current, metadata_fields);
349        let result = match runner.run(&input) {
350            Ok(r) => r,
351            Err(e) => {
352                tracing::warn!("hook '{name}' run error: {e}");
353                continue;
354            }
355        };
356        match apply_hook_result(&result, &current) {
357            Ok(out) => current = out,
358            Err(e) => tracing::warn!("hook '{name}' apply error: {e}"),
359        }
360    }
361    current
362}

445/// Compute the TextEdit needed to update `checklist-status` in a TOML metadata
446/// block to `new_status`. Returns None if not found or already correct.
447pub fn compute_toml_status_edit(content: &str, new_status: &str) -> Option<TextEdit> {
448    let block = parser::find_toml_metadata_block(content)?;
449    let lines: Vec<&str> = content.lines().collect();
450
451    for i in block.start_line..=block.end_line {
452        let line = lines.get(i)?;
453        if line.trim_start().starts_with("checklist-status") {
454            let new_line = format!("  checklist-status = \"{new_status}\"");
455            if *line == new_line {
456                return None;
457            }
458            return Some(TextEdit {
459                range: Range {
460                    start: Position {
461                        line: i as u32,
462                        character: 0,
463                    },
464                    end: Position {
465                        line: i as u32,
466                        character: line.len() as u32,
467                    },
468                },
469                new_text: new_line,
470            });
471        }
472    }
473    None
