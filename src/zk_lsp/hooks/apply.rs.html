4/// Validate that all edits in `result` are within `content` and non-overlapping.
5pub fn validate_hook_result(result: &HookResult, content: &str) -> anyhow::Result<()> {
6    let len = content.len();
7    for edit in &result.edits {
8        anyhow::ensure!(
9            edit.start_byte <= edit.end_byte,
10            "edit has start_byte ({}) > end_byte ({})",
11            edit.start_byte,
12            edit.end_byte
13        );
14        anyhow::ensure!(
15            edit.end_byte <= len,
16            "edit end_byte ({}) out of bounds (content len={})",
17            edit.end_byte,
18            len
19        );
20    }
21    // Check overlaps
22    let mut sorted: Vec<&super::types::HookTextEdit> = result.edits.iter().collect();
23    sorted.sort_by_key(|e| e.start_byte);
24    for w in sorted.windows(2) {
25        anyhow::ensure!(
26            w[0].end_byte <= w[1].start_byte,
27            "edits overlap: [{}, {}) and [{}, {})",
28            w[0].start_byte,
29            w[0].end_byte,
30            w[1].start_byte,
31            w[1].end_byte
32        );
33    }
34    Ok(())
35}

37/// Apply a `HookResult` to `content`.
38///
39/// Pipeline:
40/// 1. If `result.metadata` is non-empty: apply metadata patch.
41/// 2. If `result.edits` is non-empty: apply byte edits.
42///
43/// The Rust normalizer (`normalize_note`) is intentionally NOT called here —
44/// the Lua hook is fully responsible for formatting.
45/// For global cross-file state reconciliation, use the `reconcile` command.
46pub fn apply_hook_result(result: &HookResult, content: &str) -> anyhow::Result<String> {
47    validate_hook_result(result, content)?;
48
49    // Step 1: byte edits — applied first so their byte offsets (relative to original
50    // content) are correct before the metadata patch may shift bytes in the TOML block.
51    let after_edits = if !result.edits.is_empty() {
52        let edits_as_tuples: Vec<(usize, usize, String)> = result
53            .edits
54            .iter()
55            .map(|e| (e.start_byte, e.end_byte, e.text.clone()))
56            .collect();
57        formatting::apply_byte_edits(content, &edits_as_tuples)?
58    } else {
59        content.to_string()
60    };
61
62    // Step 2: metadata patch — uses pattern matching within the TOML block, so it is
63    // unaffected by body edits applied above (which are outside the TOML block).
64    let result_content = if !result.metadata.is_empty() {
65        formatting::apply_metadata_patch(&after_edits, &result.metadata)?
66    } else {
67        after_edits
68    };
69
70    Ok(result_content)
71}
