4/// Validate that all edits in `result` are within `content` and non-overlapping.
5pub fn validate_hook_result(result: &HookResult, content: &str) -> anyhow::Result<()> {
6    let len = content.len();
7    for edit in &result.edits {
8        anyhow::ensure!(
9            edit.start_byte <= edit.end_byte,
10            "edit has start_byte ({}) > end_byte ({})",
11            edit.start_byte,
12            edit.end_byte
13        );
14        anyhow::ensure!(
15            edit.end_byte <= len,
16            "edit end_byte ({}) out of bounds (content len={})",
17            edit.end_byte,
18            len
19        );
20    }
21    // Check overlaps
22    let mut sorted: Vec<&super::types::HookTextEdit> = result.edits.iter().collect();
23    sorted.sort_by_key(|e| e.start_byte);
24    for w in sorted.windows(2) {
25        anyhow::ensure!(
26            w[0].end_byte <= w[1].start_byte,
27            "edits overlap: [{}, {}) and [{}, {})",
28            w[0].start_byte,
29            w[0].end_byte,
30            w[1].start_byte,
31            w[1].end_byte
32        );
33    }
34    Ok(())
35}

37pub fn apply_hook_text_edits(result: &HookResult, content: &str) -> anyhow::Result<String> {
38    validate_hook_result(result, content)?;
39    if result.edits.is_empty() {
40        return Ok(content.to_string());
41    }
42    let edits_as_tuples: Vec<(usize, usize, String)> = result
43        .edits
44        .iter()
45        .map(|e| (e.start_byte, e.end_byte, e.text.clone()))
46        .collect();
47    formatting::apply_byte_edits(content, &edits_as_tuples)
48}
