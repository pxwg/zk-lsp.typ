1/// Materialize layer — v1 identity projection.
2use std::collections::HashMap;
3
4use super::eval::EvalResult;
5use super::types::{CheckboxId, CheckboxWriteback, NoteId, Value};
6
7/// v1: materialized == effective (identity projection).
8pub struct ReconcileResult {
9    #[allow(dead_code)]
10    pub materialized_meta: HashMap<(NoteId, String), Value>,
11    pub materialized_checked: HashMap<CheckboxId, CheckboxWriteback>,
12}
13
14/// v1: identity — materialized == effective.
15pub fn materialize(eval: EvalResult) -> ReconcileResult {
16    ReconcileResult {
17        materialized_meta: eval.effective_meta,
18        materialized_checked: eval.materialized_checked,
19    }
20}

26#[cfg(test)]
27mod tests {
28    use std::path::PathBuf;
29
30    use super::*;
31    use crate::reconcile::default_module::DEFAULT_MODULE;
32    use crate::reconcile::eval::eval_all;
33    use crate::reconcile::observe::WorkspaceSnapshot;
34    use crate::reconcile::parser::parse_module;
35    use crate::reconcile::types::Status;
36
37    #[test]
38    fn v1_identity() {
39        let content = "#import \"../include.typ\": *\n\
40             #let zk-metadata = toml(bytes(\n\
41             \x20 ```toml\n\
42             \x20 schema-version = 1\n\
43             \x20 title = \"Test\"\n\
44             \x20 tags = []\n\
45             \x20 checklist-status = \"done\"\n\
46             \x20 generated = false\n\
47             \x20 ```.text,\n\
48             ))\n\
49             #show: zettel.with(metadata: zk-metadata)\n\
50             \n\
51             = Test <1111111111>\n";
52        let map: std::collections::HashMap<NoteId, (PathBuf, String)> = [(
53            "1111111111".to_string(),
54            (PathBuf::from("1111111111.typ"), content.to_string()),
55        )]
56        .into_iter()
57        .collect();
58        let snap = WorkspaceSnapshot::from_note_map(&map);
59        let module = parse_module(DEFAULT_MODULE).expect("parse");
60        let eval_result = eval_all(&module, &snap);
61
62        let result = materialize(eval_result);
63
64        // checklist-status now materializes from the generic meta map.
65        assert_eq!(
66            result
67                .materialized_meta
68                .get(&("1111111111".to_string(), "checklist-status".to_string())),
69            Some(&Value::Status(Status::Done))
70        );
71    }
