14fn toml_value_to_json(v: &toml::Value) -> serde_json::Value {
15    match v {
16        toml::Value::String(s) => serde_json::Value::String(s.clone()),
17        toml::Value::Integer(n) => serde_json::Value::Number((*n).into()),
18        toml::Value::Float(f) => serde_json::Number::from_f64(*f)
19            .map(serde_json::Value::Number)
20            .unwrap_or(serde_json::Value::Null),
21        toml::Value::Boolean(b) => serde_json::Value::Bool(*b),
22        toml::Value::Array(arr) => {
23            serde_json::Value::Array(arr.iter().map(toml_value_to_json).collect())
24        }
25        toml::Value::Table(t) => {
26            let map: serde_json::Map<String, serde_json::Value> = t
27                .iter()
28                .map(|(k, v)| (k.clone(), toml_value_to_json(v)))
29                .collect();
30            serde_json::Value::Object(map)
31        }
32        toml::Value::Datetime(dt) => serde_json::Value::String(dt.to_string()),
33    }
34}

36fn merge_missing_table_defaults(target: &mut toml::Table, defaults: &toml::Table) {
37    for (key, default_value) in defaults {
38        match target.get_mut(key) {
39            Some(toml::Value::Table(target_table)) => {
40                if let toml::Value::Table(default_table) = default_value {
41                    merge_missing_table_defaults(target_table, default_table);
42                }
43            }
44            Some(_) => {}
45            None => {
46                target.insert(key.clone(), default_value.clone());
47            }
48        }
49    }
50}

62pub fn build_note_info_value(
63    id: &str,
64    path: &Path,
65    header: &NoteHeader,
66    parsed: &MetadataRecord,
67    metadata_fields: &[MetadataFieldConfig],
68) -> serde_json::Value {
69    use serde_json::{json, Map, Value};
70
71    let checklist_status_str = parsed.checklist_status.as_str();
72    let relation_str = match parsed.relation {
73        Relation::Active => "active",
74        Relation::Archived => "archived",
75        Relation::Legacy => "legacy",
76    };
77
78    let mut metadata: Map<String, Value> = Map::new();
79    metadata.insert("schema-version".into(), json!(parsed.schema_version));
80    metadata.insert("aliases".into(), json!(parsed.aliases));
81    metadata.insert(
82        "abstract".into(),
83        json!(parsed.abstract_text.as_deref().unwrap_or("")),
84    );
85    metadata.insert("keywords".into(), json!(parsed.keywords));
86    metadata.insert("generated".into(), json!(parsed.generated));
87    metadata.insert("checklist-status".into(), json!(checklist_status_str));
88    metadata.insert("relation".into(), json!(relation_str));
89    metadata.insert("relation-target".into(), json!(parsed.relation_target));
90
91    for (k, v) in &merged_extra_metadata(parsed, metadata_fields) {
92        metadata.insert(k.clone(), toml_value_to_json(v));
93    }
94
95    json!({
96        "id": id,
97        "path": path.to_string_lossy().as_ref(),
98        "title": header.title,
99        "metadata": Value::Object(metadata),
100    })
101}

177async fn build_note_values(config: &WikiConfig) -> anyhow::Result<Vec<serde_json::Value>> {
178    let mut notes = collect_note_paths(&config.note_dir).await?;
179    notes.sort_by(|(id_a, path_a), (id_b, path_b)| id_a.cmp(id_b).then_with(|| path_a.cmp(path_b)));
180    let metadata_snapshot = metadata::MetadataSnapshot::load(config)
181        .await
182        .unwrap_or_else(metadata::MetadataSnapshot::unavailable);
183    let cache_context = CacheContext::load(config).await;
184    let cache_path = config.root.join(".zk-lsp/notes-v1.json");
185    let cache = NotesCache::load(&cache_path, &cache_context).await;
186
187    const READ_CONCURRENCY: usize = 32;
188    let metadata_snapshot = &metadata_snapshot;
189    let cached_records = &cache.records;
190    let results: Vec<NoteRecordResult> = stream::iter(notes)
191        .map(move |(id, path)| async move {
192            build_note_record(
193                &id,
194                &path,
195                config,
196                metadata_snapshot,
197                cached_records.get(&id),
198            )
199            .await
200        })
201        .buffered(READ_CONCURRENCY)
202        .collect()
203        .await;
204
205    let mut records = Vec::with_capacity(results.len());
206    let mut next_cache_records = HashMap::with_capacity(results.len());
207    for result in results {
208        if let Some(warning) = &result.warning {
209            eprintln!("zk-lsp notes: warning: {warning}");
210        }
211        if let Some(fingerprint) = &result.fingerprint {
212            next_cache_records.insert(
213                result.id.clone(),
214                CachedNoteRecord {
215                    fingerprint: fingerprint.clone(),
216                    value: result.value.clone(),
217                    warning: result.warning.clone(),
218                },
219            );
220        }
221        records.push(result.value);
222    }
223    NotesCache::save(cache_path, cache_context, next_cache_records).await;
224
225    Ok(records)
226}

228async fn collect_note_paths(note_dir: &Path) -> anyhow::Result<Vec<(String, PathBuf)>> {
229    let mut notes = Vec::new();
230    let mut entries = tokio::fs::read_dir(note_dir)
231        .await
232        .map_err(|err| anyhow::anyhow!("reading note dir {}: {err}", note_dir.display()))?;
233    while let Some(entry) = entries.next_entry().await? {
234        let path = entry.path();
235        if path.extension().and_then(|e| e.to_str()) != Some("typ") {
236            continue;
237        }
238        let Some(stem) = path.file_stem().and_then(|s| s.to_str()) else {
239            continue;
240        };
241        if stem.len() == 10 && stem.chars().all(|c| c.is_ascii_digit()) {
242            notes.push((stem.to_string(), path));
243        }
244    }
245    Ok(notes)
246}

248async fn build_note_record(
249    id: &str,
250    path: &Path,
251    config: &WikiConfig,
252    metadata_snapshot: &metadata::MetadataSnapshot,
253    cached: Option<&CachedNoteRecord>,
254) -> NoteRecordResult {
255    let initial_stat = tokio::fs::metadata(path).await.ok();
256    let initial_fingerprint = initial_stat
257        .as_ref()
258        .and_then(FileFingerprint::from_metadata);
259    if let (Some(cached), Some(fingerprint)) = (cached, &initial_fingerprint) {
260        if cached.fingerprint == *fingerprint {
261            return NoteRecordResult {
262                id: id.to_string(),
263                value: cached.value.clone(),
264                warning: cached.warning.clone(),
265                fingerprint: Some(fingerprint.clone()),
266            };
267        }
268    }
269
270    let (content, stat) = match read_note_with_stat(path).await {
271        Ok(result) => result,
272        Err(err) => {
273            let message = format!("reading {}: {err}", path.display());
274            return NoteRecordResult::warning(
275                id,
276                note_error_value(id, path, &message, None),
277                message,
278                None,
279            );
280        }
281    };
282
283    let record = match metadata_snapshot.record(id) {
284        Ok(record) => record,
285        Err(err) => {
286            let message = err.to_string();
287            return NoteRecordResult::warning(
288                id,
289                note_error_value(id, path, &message, stat.as_ref()),
290                format!("{}: {message}", path.display()),
291                stat.as_ref().and_then(FileFingerprint::from_metadata),
292            );
293        }
294    };
295
296    match build_note_info_value_from_content(
297        id,
298        path,
299        &content,
300        &record,
301        &config.zk_config.metadata.fields,
302    ) {
303        Ok(mut value) => {
304            add_file_stat(&mut value, stat.as_ref());
305            NoteRecordResult::ok(
306                id,
307                value,
308                stat.as_ref().and_then(FileFingerprint::from_metadata),
309            )
310        }
311        Err(err) => {
312            let message = err.to_string();
313            NoteRecordResult::warning(
314                id,
315                note_error_value(id, path, &message, stat.as_ref()),
316                format!("{}: {message}", path.display()),
317                stat.as_ref().and_then(FileFingerprint::from_metadata),
318            )
319        }
320    }
321}

323async fn read_note_with_stat(path: &Path) -> anyhow::Result<(String, Option<std::fs::Metadata>)> {
324    let path = path.to_owned();
325    tokio::task::spawn_blocking(move || {
326        use std::io::Read;
327
328        let mut file = std::fs::File::open(path)?;
329        let stat = file.metadata().ok();
330        let mut content = String::with_capacity(
331            stat.as_ref()
332                .and_then(|metadata| usize::try_from(metadata.len()).ok())
333                .unwrap_or(0),
334        );
335        file.read_to_string(&mut content)?;
336        Ok((content, stat))
337    })
338    .await
339    .map_err(|err| anyhow::anyhow!("note reader task failed: {err}"))?
340}

376#[derive(Clone, Debug, Deserialize, Eq, PartialEq, Serialize)]
377struct FileFingerprint {
378    size: u64,
379    modified_nanos: u64,
380}
381
382impl FileFingerprint {
383    fn from_metadata(metadata: &std::fs::Metadata) -> Option<Self> {
384        let nanos = metadata
385            .modified()
386            .ok()?
387            .duration_since(UNIX_EPOCH)
388            .ok()?
389            .as_nanos();
390        Some(Self {
391            size: metadata.len(),
392            modified_nanos: nanos.min(u128::from(u64::MAX)) as u64,
393        })
394    }
395
396    async fn load(path: &Path) -> Option<Self> {
397        Self::from_metadata(&tokio::fs::metadata(path).await.ok()?)
398    }
399}

401#[derive(Clone, Debug, Deserialize, Eq, PartialEq, Serialize)]
402struct CacheContext {
403    metadata: Option<FileFingerprint>,
404    metadata_config_hash: u64,
405}
406
407impl CacheContext {
408    async fn load(config: &WikiConfig) -> Self {
409        let mut hasher = DefaultHasher::new();
410        for field in &config.zk_config.metadata.fields {
411            field.path.hash(&mut hasher);
412            format!("{:?}", field.kind).hash(&mut hasher);
413            field.default.to_string().hash(&mut hasher);
414        }
415        Self {
416            metadata: FileFingerprint::load(&metadata::metadata_path(&config.root)).await,
417            metadata_config_hash: hasher.finish(),
418        }
419    }
420}

429#[derive(Deserialize, Serialize)]
430struct NotesCache {
431    version: u32,
432    context: CacheContext,
433    records: HashMap<String, CachedNoteRecord>,
434}
435
436impl NotesCache {
437    async fn load(path: &Path, context: &CacheContext) -> Self {
438        let loaded = match tokio::fs::read(path).await {
439            Ok(bytes) => serde_json::from_slice::<Self>(&bytes).ok(),
440            Err(_) => None,
441        };
442        match loaded {
443            Some(cache) if cache.version == NOTES_CACHE_VERSION && cache.context == *context => {
444                cache
445            }
446            _ => Self {
447                version: NOTES_CACHE_VERSION,
448                context: context.clone(),
449                records: HashMap::new(),
450            },
451        }
452    }
453
454    async fn save(
455        path: PathBuf,
456        context: CacheContext,
457        records: HashMap<String, CachedNoteRecord>,
458    ) {
459        let cache = Self {
460            version: NOTES_CACHE_VERSION,
461            context,
462            records,
463        };
464        let Ok(bytes) = serde_json::to_vec(&cache) else {
465            return;
466        };
467        let Some(parent) = path.parent() else {
468            return;
469        };
470        if tokio::fs::create_dir_all(parent).await.is_err() {
471            return;
472        }
473        let temp = path.with_extension(format!("tmp-{}", std::process::id()));
474        if tokio::fs::write(&temp, bytes).await.is_ok()
475            && tokio::fs::rename(&temp, &path).await.is_err()
476        {
477            let _ = tokio::fs::remove_file(&temp).await;
478        }
479    }
480}

482fn note_error_value(
483    id: &str,
484    path: &Path,
485    message: &str,
486    stat: Option<&std::fs::Metadata>,
487) -> serde_json::Value {
488    let mut value = serde_json::json!({
489        "id": id,
490        "path": path.to_string_lossy().as_ref(),
491        "title": "",
492        "metadata": {},
493        "error": message,
494    });
495    add_file_stat(&mut value, stat);
496    value
497}

499fn add_file_stat(value: &mut serde_json::Value, stat: Option<&std::fs::Metadata>) {
500    let Some(obj) = value.as_object_mut() else {
501        return;
502    };
503    let Some(stat) = stat else {
504        return;
505    };
506    obj.insert("size".into(), serde_json::json!(stat.len()));
507    if let Ok(modified) = stat.modified() {
508        if let Ok(duration) = modified.duration_since(UNIX_EPOCH) {
509            obj.insert("mtime".into(), serde_json::json!(duration.as_secs()));
510        }
511    }
512}

514fn parse_note_info_content(id: &str, content: &str) -> anyhow::Result<NoteHeader> {
515    let header = crate::parser::parse_header(content).ok_or_else(|| {
516        anyhow::anyhow!("Failed to parse note {id}: missing or invalid title/metadata binding")
517    })?;
518    if header.id != id {
519        anyhow::bail!("Failed to parse note {id}: title ID does not match current note");
520    }
521    Ok(header)
522}
