5fn toml_value_to_json(v: &toml::Value) -> serde_json::Value {
6    match v {
7        toml::Value::String(s) => serde_json::Value::String(s.clone()),
8        toml::Value::Integer(n) => serde_json::Value::Number((*n).into()),
9        toml::Value::Float(f) => serde_json::Number::from_f64(*f)
10            .map(serde_json::Value::Number)
11            .unwrap_or(serde_json::Value::Null),
12        toml::Value::Boolean(b) => serde_json::Value::Bool(*b),
13        toml::Value::Array(arr) => {
14            serde_json::Value::Array(arr.iter().map(toml_value_to_json).collect())
15        }
16        toml::Value::Table(t) => {
17            let map: serde_json::Map<String, serde_json::Value> = t
18                .iter()
19                .map(|(k, v)| (k.clone(), toml_value_to_json(v)))
20                .collect();
21            serde_json::Value::Object(map)
22        }
23        toml::Value::Datetime(dt) => serde_json::Value::String(dt.to_string()),
24    }
25}

27pub fn build_note_info_value(
28    id: &str,
29    path: &Path,
30    header: &NoteHeader,
31    parsed: &ParsedToml,
32) -> serde_json::Value {
33    use serde_json::{json, Map, Value};
34
35    let checklist_status_str = parsed.checklist_status.as_str();
36    let relation_str = match parsed.relation {
37        Relation::Active => "active",
38        Relation::Archived => "archived",
39        Relation::Legacy => "legacy",
40    };
41
42    let mut metadata: Map<String, Value> = Map::new();
43    metadata.insert("schema-version".into(), json!(parsed.schema_version));
44    metadata.insert("aliases".into(), json!(parsed.aliases));
45    metadata.insert(
46        "abstract".into(),
47        json!(parsed.abstract_text.as_deref().unwrap_or("")),
48    );
49    metadata.insert("keywords".into(), json!(parsed.keywords));
50    metadata.insert("generated".into(), json!(parsed.generated));
51    metadata.insert("checklist-status".into(), json!(checklist_status_str));
52    metadata.insert("relation".into(), json!(relation_str));
53    metadata.insert("relation-target".into(), json!(parsed.relation_target));
54
55    for (k, v) in &parsed.extra {
56        metadata.insert(k.clone(), toml_value_to_json(v));
57    }
58
59    json!({
60        "id": id,
61        "path": path.to_string_lossy().as_ref(),
62        "title": header.title,
63        "metadata": Value::Object(metadata),
64    })
65}

67pub fn build_note_info_json(
68    id: &str,
69    path: &Path,
70    header: &NoteHeader,
