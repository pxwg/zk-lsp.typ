8fn toml_value_to_json(v: &toml::Value) -> serde_json::Value {
9    match v {
10        toml::Value::String(s) => serde_json::Value::String(s.clone()),
11        toml::Value::Integer(n) => serde_json::Value::Number((*n).into()),
12        toml::Value::Float(f) => serde_json::Number::from_f64(*f)
13            .map(serde_json::Value::Number)
14            .unwrap_or(serde_json::Value::Null),
15        toml::Value::Boolean(b) => serde_json::Value::Bool(*b),
16        toml::Value::Array(arr) => {
17            serde_json::Value::Array(arr.iter().map(toml_value_to_json).collect())
18        }
19        toml::Value::Table(t) => {
20            let map: serde_json::Map<String, serde_json::Value> = t
21                .iter()
22                .map(|(k, v)| (k.clone(), toml_value_to_json(v)))
23                .collect();
24            serde_json::Value::Object(map)
25        }
26        toml::Value::Datetime(dt) => serde_json::Value::String(dt.to_string()),
27    }
28}

30fn merge_missing_table_defaults(target: &mut toml::Table, defaults: &toml::Table) {
31    for (key, default_value) in defaults {
32        match target.get_mut(key) {
33            Some(toml::Value::Table(target_table)) => {
34                if let toml::Value::Table(default_table) = default_value {
35                    merge_missing_table_defaults(target_table, default_table);
36                }
37            }
38            Some(_) => {}
39            None => {
40                target.insert(key.clone(), default_value.clone());
41            }
42        }
43    }
44}

46fn merged_extra_metadata(
47    parsed: &ParsedToml,
48    metadata_fields: &[MetadataFieldConfig],
49) -> toml::Table {
50    let mut merged = parsed.extra.clone();
51    let defaults = metadata_defaults_table(metadata_fields);
52    merge_missing_table_defaults(&mut merged, &defaults);
53    merged
54}

56pub fn build_note_info_value(
57    id: &str,
58    path: &Path,
59    header: &NoteHeader,
60    parsed: &ParsedToml,
61    metadata_fields: &[MetadataFieldConfig],
62) -> serde_json::Value {
63    use serde_json::{json, Map, Value};
64
65    let checklist_status_str = parsed.checklist_status.as_str();
66    let relation_str = match parsed.relation {
67        Relation::Active => "active",
68        Relation::Archived => "archived",
69        Relation::Legacy => "legacy",
70    };
71
72    let mut metadata: Map<String, Value> = Map::new();
73    metadata.insert("schema-version".into(), json!(parsed.schema_version));
74    metadata.insert("aliases".into(), json!(parsed.aliases));
75    metadata.insert(
76        "abstract".into(),
77        json!(parsed.abstract_text.as_deref().unwrap_or("")),
78    );
79    metadata.insert("keywords".into(), json!(parsed.keywords));
80    metadata.insert("generated".into(), json!(parsed.generated));
81    metadata.insert("checklist-status".into(), json!(checklist_status_str));
82    metadata.insert("relation".into(), json!(relation_str));
83    metadata.insert("relation-target".into(), json!(parsed.relation_target));
84
85    for (k, v) in &merged_extra_metadata(parsed, metadata_fields) {
86        metadata.insert(k.clone(), toml_value_to_json(v));
87    }
88
89    json!({
90        "id": id,
91        "path": path.to_string_lossy().as_ref(),
92        "title": header.title,
93        "metadata": Value::Object(metadata),
94    })
95}

113pub fn build_note_info_json(

115    path: &Path,
116    header: &NoteHeader,
117    parsed: &ParsedToml,
118    metadata_fields: &[MetadataFieldConfig],

120) -> anyhow::Result<String> {

129pub async fn build_single_note_info_json(id: &str, config: &WikiConfig) -> anyhow::Result<String> {
130    let path = config.note_dir.join(format!("{id}.typ"));
131    if !path.exists() {
132        anyhow::bail!("Note {id} not found at {}", path.display());
133    }
134    let content = tokio::fs::read_to_string(&path)
135        .await
136        .map_err(|err| anyhow::anyhow!("reading {}: {err}", path.display()))?;
137    let (header, parsed_toml) = parse_note_info_content(id, &content)?;
138    build_note_info_json(
139        id,
140        &path,
141        &header,
142        &parsed_toml,
143        &config.zk_config.metadata.fields,
144        &content,
145    )
146}

148pub async fn build_notes_json(config: &WikiConfig) -> anyhow::Result<String> {
149    let mut notes = collect_note_paths(&config.note_dir).await?;
150    notes.sort_by(|(id_a, path_a), (id_b, path_b)| id_a.cmp(id_b).then_with(|| path_a.cmp(path_b)));
151
152    let mut records = Vec::with_capacity(notes.len());
153    for (id, path) in notes {
154        records.push(build_note_record(&id, &path, config).await);
155    }
156
157    Ok(serde_json::to_string_pretty(&records)?)
158}

160async fn collect_note_paths(note_dir: &Path) -> anyhow::Result<Vec<(String, PathBuf)>> {
161    let mut notes = Vec::new();
162    let mut entries = tokio::fs::read_dir(note_dir)
163        .await
164        .map_err(|err| anyhow::anyhow!("reading note dir {}: {err}", note_dir.display()))?;
165    while let Some(entry) = entries.next_entry().await? {
166        let path = entry.path();
167        if path.extension().and_then(|e| e.to_str()) != Some("typ") {
168            continue;
169        }
170        let Some(stem) = path.file_stem().and_then(|s| s.to_str()) else {
171            continue;
172        };
173        if stem.len() == 10 && stem.chars().all(|c| c.is_ascii_digit()) {
174            notes.push((stem.to_string(), path));
175        }
176    }
177    Ok(notes)
178}

180async fn build_note_record(id: &str, path: &Path, config: &WikiConfig) -> serde_json::Value {
181    let stat = tokio::fs::metadata(path).await.ok();
182    let content = match tokio::fs::read_to_string(path).await {
183        Ok(content) => content,
184        Err(err) => {
185            let message = format!("reading {}: {err}", path.display());
186            eprintln!("zk-lsp notes: warning: {message}");
187            return note_error_value(id, path, &message, stat.as_ref());
188        }
189    };
190
191    match build_note_info_value_from_content(id, path, &content, &config.zk_config.metadata.fields)
192    {
193        Ok(mut value) => {
194            add_file_stat(&mut value, stat.as_ref());
195            value
196        }
197        Err(err) => {
198            let message = err.to_string();
199            eprintln!("zk-lsp notes: warning: {}: {message}", path.display());
200            note_error_value(id, path, &message, stat.as_ref())
201        }
202    }
203}

205fn note_error_value(
206    id: &str,
207    path: &Path,
208    message: &str,
209    stat: Option<&std::fs::Metadata>,
210) -> serde_json::Value {
211    let mut value = serde_json::json!({
212        "id": id,
213        "path": path.to_string_lossy().as_ref(),
214        "title": "",
215        "metadata": {},
216        "error": message,
217    });
218    add_file_stat(&mut value, stat);
219    value
220}

222fn add_file_stat(value: &mut serde_json::Value, stat: Option<&std::fs::Metadata>) {
223    let Some(obj) = value.as_object_mut() else {
224        return;
225    };
226    let Some(stat) = stat else {
227        return;
228    };
229    obj.insert("size".into(), serde_json::json!(stat.len()));
230    if let Ok(modified) = stat.modified() {
231        if let Ok(duration) = modified.duration_since(UNIX_EPOCH) {
232            obj.insert("mtime".into(), serde_json::json!(duration.as_secs()));
233        }
234    }
235}

237fn parse_note_info_content(id: &str, content: &str) -> anyhow::Result<(NoteHeader, ParsedToml)> {
238    let Some(block) = crate::parser::find_toml_metadata_block(content) else {
239        anyhow::bail!("Failed to parse note {id} (may be legacy format; run zk-lsp migrate first)");
240    };
241    let parsed_toml = crate::parser::parse_toml_metadata(&block.toml_content).ok_or_else(|| {
242        let detail = match block.toml_content.parse::<toml::Value>() {
243            Ok(_) => "expected TOML metadata table".to_string(),
244            Err(err) => err.to_string(),
245        };
246        anyhow::anyhow!("Failed to parse TOML metadata for note {id}: {detail}")
247    })?;
248    let header = crate::parser::parse_header(content).ok_or_else(|| {
249        anyhow::anyhow!(
250            "Failed to parse note {id} (may be legacy format; run zk-lsp migrate first)"
251        )
252    })?;
253    Ok((header, parsed_toml))
254}
