1/// Build a positioned dependency graph from note content.
2///
3/// Each `RefItem` checklist entry produces one `CycleEdgeOccurrence` per `RefTarget`,
4/// carrying the file location needed for cycle error reporting.
5use std::collections::HashMap;
6use std::path::PathBuf;
7
8use crate::parser;
9
10#[derive(Debug, Clone)]
11pub struct CycleEdgeOccurrence {
12    pub from_note_id: String,
13    pub to_note_id: String,
14    pub file_path: PathBuf,
15    pub line: usize,     // 0-based line index
16    pub byte_start: u32, // byte offset of '@' within the line
17    pub byte_end: u32,   // byte offset past the last digit
18    #[allow(dead_code)]
19    pub line_text: String, // full line text for display
20}
21
22pub struct DependencyGraph {
23    pub nodes: Vec<String>,
24    pub adj: HashMap<String, Vec<String>>,
25    pub occurrences: Vec<CycleEdgeOccurrence>,
26}
27
28/// Build a `DependencyGraph` from a map of `id → (path, content)`.
29///
30/// Every `RefItem` checklist entry contributes one directed edge per `RefTarget`:
31/// `from_note_id → to_note_id`. Duplicate edges in `adj` are deduplicated; all
32/// positioned occurrences are retained for error reporting.
33pub fn build_dependency_graph(notes: &HashMap<String, (PathBuf, String)>) -> DependencyGraph {
34    let mut adj: HashMap<String, Vec<String>> = HashMap::new();
35    let mut occurrences: Vec<CycleEdgeOccurrence> = Vec::new();
36
37    for (from_id, (path, content)) in notes {
38        adj.entry(from_id.clone()).or_default();
39        let items = parser::parse_checklist_items(content);
40        let lines: Vec<&str> = content.lines().collect();
41
42        for item in items {
43            if let parser::ChecklistItemKind::Ref { targets } = item.kind {
44                let line_text = lines.get(item.line_idx).copied().unwrap_or("").to_string();
45                for target in targets {
46                    occurrences.push(CycleEdgeOccurrence {
47                        from_note_id: from_id.clone(),
48                        to_note_id: target.target_id.clone(),
49                        file_path: path.clone(),
50                        line: item.line_idx,
51                        byte_start: target.byte_start,
52                        byte_end: target.byte_end,
53                        line_text: line_text.clone(),
54                    });
55                    let targets_list = adj.entry(from_id.clone()).or_default();
56                    if !targets_list.contains(&target.target_id) {
57                        targets_list.push(target.target_id.clone());
58                    }
59                    // Ensure the dependency node appears in adj (even if it has no outgoing edges)
60                    adj.entry(target.target_id.clone()).or_default();
61                }
62            }
63        }
64    }
65
66    let nodes = adj.keys().cloned().collect();
67    DependencyGraph {
68        nodes,
69        adj,
70        occurrences,
71    }
72}
