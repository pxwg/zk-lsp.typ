47    async fn document_in_scope(&self, uri: &Url) -> bool {
48        let Ok(path) = uri.to_file_path() else {
49            return false;
50        };
51        let config = self.current_config().await;
52        is_workspace_document_path(&config, &path)
53    }
54
55    async fn publish_diagnostics(&self, uri: Url, content: &str) {
56        let file_path = uri.to_file_path().unwrap_or_default();
57        let config = self.current_config().await;
58        if !is_workspace_document_path(&config, &file_path) {
59            self.client.publish_diagnostics(uri, Vec::new(), None).await;
60            return;
61        }
62        if is_metadata_path(&config, &file_path) {
63            self.client
64                .publish_diagnostics(
65                    uri,
66                    diagnostics::get_metadata_index_diagnostics_with_context(
67                        content,
68                        &config.zk_config.metadata.fields,
69                        Some(&config.note_dir),
70                    ),
71                    None,
72                )
73                .await;
74            return;
75        }
76
77        let mut diags = diagnostics::get_diagnostics(content, &self.index, uri.path());
78        diags.extend(diagnostics::get_schema_diagnostics_for_note(
79            content,
80            path_note_id(&file_path),
81        ));
82        diags.extend(get_note_metadata_diagnostics(content, &config).await);
83        if let Ok(reconcile_diags) =
84            reconcile::collect_diagnostics(&config, Some((&file_path, content))).await
85        {
86            diags.extend(diagnostics::get_reconcile_diagnostics(
87                content,
88                &file_path,
89                &reconcile_diags,
90            ));
91        }
92        if let Some(d) = diagnostics::get_orphan_diagnostic(content, uri.path(), &self.index) {
93            diags.push(d);
94        }
95        self.client.publish_diagnostics(uri, diags, None).await;
96    }
97
98    async fn read_document(&self, uri: &Url) -> Option<String> {
99        if let Some(content) = self.open_documents.read().await.get(uri).cloned() {
100            return Some(content);
101        }
102
103        uri.to_file_path()
104            .ok()
105            .and_then(|path| std::fs::read_to_string(path).ok())
106    }
107
108    async fn read_metadata_document(&self, config: &WikiConfig) -> Option<(Url, String)> {
109        let path = metadata::metadata_path(&config.root);
110        let uri = Url::from_file_path(&path).ok()?;
111        let content = self.read_document(&uri).await?;
112        Some((uri, content))
113    }
114}

167    async fn initialized(&self, _params: InitializedParams) {
168        info!("server initialized, building index…");
169        let index = Arc::clone(&self.index);
170        let config = Arc::clone(&self.config);
171        let client = self.client.clone();
172        let open_documents = Arc::clone(&self.open_documents);
173
174        tokio::spawn(async move {
175            match index.rebuild_full().await {
176                Ok(n) => {
177                    info!("index built: {n} notes");
178                    for (uri, content) in open_documents.read().await.iter() {
179                        let file_path = uri.to_file_path().unwrap_or_default();
180                        let current_config = config.read().await.clone();
181                        if !is_workspace_document_path(&current_config, &file_path) {
182                            client
183                                .publish_diagnostics(uri.clone(), Vec::new(), None)
184                                .await;
185                            continue;
186                        }
187                        if is_metadata_path(&current_config, &file_path) {
188                            client
189                                .publish_diagnostics(
190                                    uri.clone(),
191                                    diagnostics::get_metadata_index_diagnostics_with_context(
192                                        content,
193                                        &current_config.zk_config.metadata.fields,
194                                        Some(&current_config.note_dir),
195                                    ),
196                                    None,
197                                )
198                                .await;
199                            continue;
200                        }
201                        let mut diags = diagnostics::get_diagnostics(content, &index, uri.path());
202                        diags.extend(diagnostics::get_schema_diagnostics_for_note(
203                            content,
204                            path_note_id(&file_path),
205                        ));
206                        diags.extend(get_note_metadata_diagnostics(content, &current_config).await);
207                        if let Ok(reconcile_diags) = reconcile::collect_diagnostics(
208                            &current_config,
209                            Some((&file_path, content)),
210                        )
211                        .await
212                        {
213                            diags.extend(diagnostics::get_reconcile_diagnostics(
214                                content,
215                                &file_path,
216                                &reconcile_diags,
217                            ));
218                        }
219                        if let Some(d) =
220                            diagnostics::get_orphan_diagnostic(content, uri.path(), &index)
221                        {
222                            diags.push(d);
223                        }
224                        client.publish_diagnostics(uri.clone(), diags, None).await;
225                    }
226                    // Tell the client to re-request inlay hints now that the index is ready.
227                    // Ask the client to re-fetch all inlay hints now that the
228                    // index is populated.
229                    let _ = client.inlay_hint_refresh().await;
230                }
231                Err(e) => error!("index build failed: {e}"),
232            }
233            // Start filesystem watcher
234            if let Err(e) = watcher::start_watcher(config, index).await {
235                error!("watcher start failed: {e}");
236            }
237        });
238    }

248    async fn did_open(&self, params: DidOpenTextDocumentParams) {
249        let uri = params.text_document.uri.clone();
250        let content = params.text_document.text;
251        if !self.document_in_scope(&uri).await {
252            self.open_documents.write().await.remove(&uri);
253            self.client.publish_diagnostics(uri, Vec::new(), None).await;
254            return;
255        }
256
257        self.open_documents
258            .write()
259            .await
260            .insert(uri.clone(), content.clone());
261        // Update index for this file
262        if let Ok(path) = uri.to_file_path() {
263            let config = self.current_config().await;
264            if !is_scoped_note_path(&config, &path) {
265                self.publish_diagnostics(uri, &content).await;
266                return;
267            }
268            let _ = self.index.update_file(&path).await;
269        }
270        self.publish_diagnostics(uri, &content).await;
271        // Tell VS Code to re-request inlay hints for this newly opened file.
272        // Without this, VS Code caches the empty first response and never retries.
273        let _ = self.client.inlay_hint_refresh().await;
274    }
275
276    async fn did_change(&self, params: DidChangeTextDocumentParams) {
277        let uri = params.text_document.uri;
278        if !self.document_in_scope(&uri).await {
279            self.open_documents.write().await.remove(&uri);
280            self.client.publish_diagnostics(uri, Vec::new(), None).await;
281            return;
282        }
283
284        let initial_content = self.read_document(&uri).await.unwrap_or_default();
285        let updated_content = {
286            let mut documents = self.open_documents.write().await;
287            let content = documents.entry(uri.clone()).or_insert(initial_content);
288            apply_content_changes(content, &params.content_changes);
289            content.clone()
290        };
291
292        self.publish_diagnostics(uri, &updated_content).await;
293    }
294
295    async fn did_save(&self, params: DidSaveTextDocumentParams) {
296        let uri = params.text_document.uri.clone();
297        if !self.document_in_scope(&uri).await {
298            self.open_documents.write().await.remove(&uri);
299            self.client.publish_diagnostics(uri, Vec::new(), None).await;
300            return;
301        }
302
303        let content = match params.text {
304            Some(t) => t,
305            None => match uri
306                .to_file_path()
307                .ok()
308                .and_then(|p| std::fs::read_to_string(p).ok())
309            {
310                Some(t) => t,
311                None => return,
312            },
313        };
314
315        self.open_documents
316            .write()
317            .await
318            .insert(uri.clone(), content.clone());
319
320        // Update index
321        if let Ok(path) = uri.to_file_path() {
322            let config = self.current_config().await;
323            if is_metadata_path(&config, &path) {
324                self.publish_diagnostics(uri.clone(), &content).await;
325                let _ = self.index.rebuild_full().await;
326                let _ = self.client.inlay_hint_refresh().await;
327                return;
328            }
329            if !is_scoped_note_path(&config, &path) {
330                self.publish_diagnostics(uri.clone(), &content).await;
331                return;
332            }
333            let _ = self.index.update_file(&path).await;
334        }
335
336        // Publish diagnostics for the saved file
337        self.publish_diagnostics(uri.clone(), &content).await;
338    }
339
340    async fn did_close(&self, params: DidCloseTextDocumentParams) {
341        let uri = params.text_document.uri;
342        self.open_documents.write().await.remove(&uri);
343        self.client.publish_diagnostics(uri, Vec::new(), None).await;
344    }
345
346    async fn did_change_watched_files(&self, params: DidChangeWatchedFilesParams) {
347        for change in params.changes {
348            let uri = change.uri.clone();
349            if let Ok(path) = uri.to_file_path() {
350                let config = self.current_config().await;
351                if !is_workspace_document_path(&config, &path) {
352                    self.client.publish_diagnostics(uri, Vec::new(), None).await;
353                    continue;
354                }
355                if is_metadata_path(&config, &path) {
356                    if let Ok(content) = tokio::fs::read_to_string(&path).await {
357                        self.publish_diagnostics(uri, &content).await;
358                    }
359                    let _ = self.index.rebuild_full().await;
360                    let _ = self.client.inlay_hint_refresh().await;
361                    continue;
362                }
363
364                match change.typ {
365                    FileChangeType::CREATED | FileChangeType::CHANGED => {
366                        let _ = self.index.update_file(&path).await;
367                        if let Ok(content) = tokio::fs::read_to_string(&path).await {
368                            self.publish_diagnostics(uri, &content).await;
369                        }
370                    }
371                    FileChangeType::DELETED => {
372                        self.index.remove_by_path(&path);
373                    }
374                    _ => {}
375                }
376            }
377        }
378    }

384    async fn goto_definition(
385        &self,
386        params: GotoDefinitionParams,
387    ) -> LspResult<Option<GotoDefinitionResponse>> {
388        let uri = &params.text_document_position_params.text_document.uri;
389        if !self.document_in_scope(uri).await {
390            return Ok(None);
391        }
392
393        let position = params.text_document_position_params.position;
394        let content = self.read_document(uri).await.unwrap_or_default();
395        let config = self.current_config().await;
396        let path = uri.to_file_path().unwrap_or_default();
397
398        let location = if is_metadata_path(&config, &path) {
399            definition::get_metadata_definition(&content, position, &self.index)
400        } else {
401            self.read_metadata_document(&config).await.and_then(
402                |(metadata_uri, metadata_content)| {
403                    definition::get_note_definition(
404                        &content,
405                        position,
406                        &metadata_uri,
407                        &metadata_content,
408                    )
409                },
410            )
411        };
412
413        Ok(location.map(GotoDefinitionResponse::Scalar))
414    }

576    async fn execute_command(&self, params: ExecuteCommandParams) -> LspResult<Option<Value>> {
577        match params.command.as_str() {
578            "zk.generateLinkTyp" => {
579                let config = self.current_config().await;
580                match link_gen::generate_link_typ(&config).await {
581                    Ok(()) => info!("link.typ regenerated"),
582                    Err(e) => error!("generate_link_typ: {e}"),
583                }
584            }
585            "zk.newNote" => {
586                let config = self.current_config().await;
587                if self.metadata_document_is_dirty(&config).await {
588                    self.client
589                        .show_message(
590                            MessageType::WARNING,
591                            "Save metadata.toml before running zk.newNote",
592                        )
593                        .await;
594                    return Ok(None);
595                }
596                match note_ops::create_note(&config, None, &MetaOverrides::new(), None, None).await
597                {
598                    Ok(path) => {
599                        info!("created note: {}", path.display());
600                        let uri = Url::from_file_path(&path).ok();
601                        if let Some(uri) = uri {
602                            self.client
603                                .show_message(MessageType::INFO, format!("Created: {uri}"))
604                                .await;
605                        }
606                    }
607                    Err(e) => error!("create_note: {e}"),
608                }
609            }
610            "zk.removeNote" => {
611                if let Some(id) = params.arguments.first().and_then(|v| v.as_str()) {
612                    let config = self.current_config().await;
613                    if self.metadata_document_is_dirty(&config).await {
614                        self.client
615                            .show_message(
616                                MessageType::WARNING,
617                                "Save metadata.toml before running zk.removeNote",
618                            )
619                            .await;
620                        return Ok(None);
621                    }
622                    match note_ops::delete_note(id, &config).await {
623                        Ok(()) => info!("deleted note {id}"),
624                        Err(e) => error!("delete_note: {e}"),
625                    }
626                }
627            }
628            "zk.exportContext" => {
629                let id = params
630                    .arguments
631                    .first()
632                    .and_then(|v| v.as_str())
633                    .unwrap_or("")
634                    .to_string();
635                let depth = params
636                    .arguments
637                    .get(1)
638                    .and_then(|v| v.as_u64())
639                    .unwrap_or(2) as usize;
640                let inverse = params
641                    .arguments
642                    .get(2)
643                    .and_then(|v| v.as_bool())
644                    .unwrap_or(false);
645                let simple = params
646                    .arguments
647                    .get(3)
648                    .and_then(|v| v.as_bool())
649                    .unwrap_or(false);
650                let config = self.current_config().await;
651                match crate::context_export::export_context(&id, depth, inverse, simple, &config)
652                    .await
653                {
654                    Ok(text) => return Ok(Some(Value::String(text))),
655                    Err(e) => error!("exportContext: {e}"),
656                }
657            }
658            cmd => info!("unhandled command: {cmd}"),
659        }
660        Ok(None)
661    }

671fn is_scoped_note_path(config: &WikiConfig, path: &std::path::Path) -> bool {
672    let note_dir = absolute_path(&config.note_dir);
673    let path = absolute_path(path);
674
675    path.extension().and_then(|e| e.to_str()) == Some("typ") && path.starts_with(&note_dir)
676}
677
678fn is_metadata_path(config: &WikiConfig, path: &std::path::Path) -> bool {
679    absolute_path(path) == absolute_path(&config.root.join("metadata.toml"))
680}

682fn is_workspace_document_path(config: &WikiConfig, path: &std::path::Path) -> bool {

686async fn get_note_metadata_diagnostics(content: &str, config: &WikiConfig) -> Vec<Diagnostic> {
687    let Some(header) = parser::parse_header(content) else {
688        return Vec::new();
689    };
690    match metadata::read_record_table(config, &header.id).await {
691        Ok(table) => {
692            diagnostics::note_metadata_record_message(&table, &config.zk_config.metadata.fields)
693                .map(|message| note_metadata_diagnostic(content, &header, message))
694                .into_iter()
695                .collect()
696        }
697        Err(err) => {
698            let message = if err.chain().any(|cause| {
699                cause
700                    .to_string()
701                    .contains("Missing metadata for current note")
702            }) {
703                "Missing metadata for current note"
704            } else {
705                "Invalid metadata for current note"
706            };
707            vec![note_metadata_diagnostic(content, &header, message)]
708        }
709    }
710}

712fn note_metadata_diagnostic(
713    content: &str,
714    header: &parser::NoteHeader,
715    message: &str,
716) -> Diagnostic {
717    let line_text = content.lines().nth(header.title_line_idx).unwrap_or("");
718    Diagnostic {
719        range: Range {
720            start: Position {
721                line: header.title_line_idx as u32,
722                character: 0,
723            },
724            end: Position {
725                line: header.title_line_idx as u32,
726                character: line_text.len() as u32,
727            },
728        },
729        severity: Some(DiagnosticSeverity::ERROR),
730        source: Some("zk-lsp".into()),
731        message: message.to_string(),
732        ..Default::default()
733    }
734}

742fn metadata_document_is_dirty(open_documents: &HashMap<Url, String>, config: &WikiConfig) -> bool {
743    let path = metadata::metadata_path(&config.root);
744    let Some(uri) = Url::from_file_path(&path).ok() else {
745        return false;
746    };
747    let Some(open_content) = open_documents.get(&uri) else {
748        return false;
749    };
750    match std::fs::read_to_string(path) {
751        Ok(disk_content) => disk_content != *open_content,
752        Err(_) => true,
753    }
754}

766fn apply_content_changes(content: &mut String, changes: &[TextDocumentContentChangeEvent]) {
767    for change in changes {
768        match change.range {
769            Some(range) => {
770                let start = position_to_byte_offset(content, range.start);
771                let end = position_to_byte_offset(content, range.end);
772                content.replace_range(start..end, &change.text);
773            }
774            None => {
775                *content = change.text.clone();
776            }
777        }
778    }
779}
780
781fn position_to_byte_offset(content: &str, position: Position) -> usize {
782    let line_start = if position.line == 0 {
783        0
784    } else {
785        match content
786            .match_indices('\n')
787            .nth(position.line.saturating_sub(1) as usize)
788            .map(|(idx, _)| idx + 1)
789        {
790            Some(offset) => offset,
791            None => content.len(),
792        }
793    };
794
795    let line_end = content[line_start..]
796        .find('\n')
797        .map(|idx| line_start + idx)
798        .unwrap_or(content.len());
799    let line = &content[line_start..line_end];
800
801    let mut utf16_units = 0u32;
802    for (byte_idx, ch) in line.char_indices() {
803        if utf16_units >= position.character {
804            return line_start + byte_idx;
805        }
806        utf16_units += ch.len_utf16() as u32;
807    }
808
809    line_end
810}
