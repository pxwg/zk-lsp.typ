1use anyhow::{Context, Result};
2use tokio::fs;
3use tokio::process::Command;

5use crate::config::WikiConfig;

8/// Minimal `include.typ` placed at the wiki root.
9///
10/// Provides only what zk-lsp itself requires:
11///   - `zettel`    — render function called by every note
12///   - `zk_entry` — called by `link.typ` to include each note
13///
14/// Extend this file freely with your own imports and styling.
15const INCLUDE_TYP: &str = r#"// zk-lsp include — extend with your own imports and styling
16
17/// Render a single zettel.
18/// `metadata` is the TOML-parsed table from the note header.
19#let zettel(metadata: none, body) = body
20
21/// Include one note into the compiled document.
22/// Called automatically by link.typ.
23#let zk_entry(id, path) = {
24  include path
25}
26"#;
27
28/// Human-editable entry point for the compiled wiki document.
29/// Import link.typ (auto-generated) here and add any manual content around it.
30const INDEX_TYP: &str = r#"// index.typ — editable entry point for your wiki.
31// Compile this file with Typst to produce the full document.
32//
33// link.typ is auto-generated by zk-lsp and lists every note.
34// Edit *this* file freely; do not edit link.typ by hand.
35
36#import "include.typ": *
37
38#include "link.typ"
39"#;
40
41const GITIGNORE: &str = "*.pdf\n";
42
43/// Initialise a fresh wiki at `config.root`.
44///
45/// Steps:
46///   1. Create root + `note/` directory.
47///   2. Write `include.typ` and an empty `link.typ`.
48///   3. Write `.gitignore`.
49///   4. `git init` (if git is available; non-fatal on failure).
50///   5. Create one sample note via `note_ops::create_note`.
51pub async fn init_wiki(config: &WikiConfig) -> Result<()> {
52    let root = &config.root;
53
54    // 1. Directories
55    fs::create_dir_all(&config.note_dir)
56        .await
57        .with_context(|| format!("creating note dir {}", config.note_dir.display()))?;
58
59    // 2. include.typ — skip if already present
60    let include_path = root.join("include.typ");
61    if !include_path.exists() {
62        fs::write(&include_path, INCLUDE_TYP)
63            .await
64            .with_context(|| format!("writing {}", include_path.display()))?;
65        eprintln!("  created  include.typ");
66    } else {
67        eprintln!("  exists   include.typ (skipped)");
68    }
69
70    // index.typ — human-editable entry point
71    let index_path = root.join("index.typ");
72    if !index_path.exists() {
73        fs::write(&index_path, INDEX_TYP)
74            .await
75            .with_context(|| format!("writing {}", index_path.display()))?;
76        eprintln!("  created  index.typ");
77    } else {
78        eprintln!("  exists   index.typ (skipped)");
79    }
80
81    // link.typ is created implicitly by note_ops::create_note → link_gen::add_entry,
82    // but we touch it early so `git init` picks it up in the initial commit listing.
83    let link_path = &config.link_file;
84    if !link_path.exists() {
85        fs::write(link_path, "#import \"include.typ\": *\n// Auto-generated by zk-lsp — do not edit manually.\n// Run `zk-lsp generate` to rebuild.\n\n")
86            .await
87            .with_context(|| format!("writing {}", link_path.display()))?;
88        eprintln!("  created  link.typ");
89    } else {
90        eprintln!("  exists   link.typ (skipped)");
91    }
92
93    // 3. .gitignore
94    let gitignore_path = root.join(".gitignore");
95    if !gitignore_path.exists() {
96        fs::write(&gitignore_path, GITIGNORE)
97            .await
98            .with_context(|| format!("writing {}", gitignore_path.display()))?;
99        eprintln!("  created  .gitignore");
100    } else {
101        eprintln!("  exists   .gitignore (skipped)");
102    }
103
104    // 4. git init (best-effort)
105    let git_dir = root.join(".git");
106    if !git_dir.exists() {
107        match Command::new("git").arg("init").arg(root).output().await {
108            Ok(out) if out.status.success() => eprintln!("  git init  {}", root.display()),
109            Ok(out) => eprintln!(
110                "  git init failed: {}",
111                String::from_utf8_lossy(&out.stderr).trim()
112            ),
113            Err(e) => eprintln!("  git not found, skipping: {e}"),
114        }
115    } else {
116        eprintln!("  exists   .git (skipped)");
117    }
118
119    // 5. Sample note
120    let note_path = note_ops::create_note(config).await?;
121    eprintln!("  created  {}", note_path.display());
122
123    eprintln!("\nWiki initialised at {}", root.display());
124    eprintln!("Next: open {} and start writing.", note_path.display());
125    Ok(())
126}
