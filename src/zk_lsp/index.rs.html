46impl NoteIndex {
47    pub fn new(config: Arc<RwLock<WikiConfig>>) -> Self {
48        NoteIndex {

50            backlinks: Arc::new(DashMap::new()),
51            config,
52        }

55    /// Rebuild the full index by scanning all notes in note_dir.
56    pub async fn rebuild_full(&self) -> Result<usize> {
57        self.notes.clear();
58        self.backlinks.clear();
59
60        let config = {
61            let config = self.config.read().await;
62            config.clone()
63        };
64        let note_dir = config.note_dir.clone();
65        let metadata_snapshot = metadata::MetadataSnapshot::load(&config)
66            .await
67            .unwrap_or_else(metadata::MetadataSnapshot::unavailable);
68        let mut entries = fs::read_dir(&note_dir).await?;
69        let mut paths = Vec::new();
70        while let Some(entry) = entries.next_entry().await? {
71            let path = entry.path();
72            if path.extension().and_then(|e| e.to_str()) == Some("typ") {
73                if let Some(stem) = path.file_stem().and_then(|s| s.to_str()) {
74                    if stem.len() == 10 && stem.chars().all(|c| c.is_ascii_digit()) {
75                        paths.push(path);
76                    }
77                }
78            }
79        }
80
81        for path in &paths {
82            let _ = self
83                .index_file_with_metadata(path, &metadata_snapshot)
84                .await;
85        }
86
87        Ok(self.notes.len())
88    }
89
90    /// Update a single file in the index.
91    pub async fn update_file(&self, path: &Path) -> Result<()> {
92        // Remove old backlinks contributed by this file
93        self.remove_backlinks_from(path);
94        self.index_file(path).await
95    }
96
97    /// Remove a note from the index by its path.
98    pub fn remove_by_path(&self, path: &Path) {
99        if let Some(stem) = path.file_stem().and_then(|s| s.to_str()) {
100            self.notes.remove(stem);
101        }
102        self.remove_backlinks_from(path);
103    }

105    pub fn get(&self, id: &str) -> Option<NoteInfo> {
106        self.notes.get(id).map(|r| r.clone())
107    }

109    /// Simple fuzzy search over title, aliases, keywords.
110    pub fn search(&self, query: &str) -> Vec<NoteInfo> {
111        let q = query.to_lowercase();
112        self.notes
113            .iter()
114            .filter(|entry| {
115                let n = entry.value();
116                n.title.to_lowercase().contains(&q)
117                    || n.id.contains(&q)
118                    || n.aliases.iter().any(|a| a.to_lowercase().contains(&q))
119                    || n.keywords.iter().any(|k| k.to_lowercase().contains(&q))
120                    || n.abstract_text
121                        .as_deref()
122                        .map(|a| a.to_lowercase().contains(&q))
123                        .unwrap_or(false)
124            })
125            .map(|entry| entry.value().clone())
126            .collect()
127    }

129    /// Get all backlink locations for an ID.

141    async fn index_file(&self, path: &Path) -> Result<()> {
142        let config = self.config.read().await.clone();
143        let metadata_snapshot = metadata::MetadataSnapshot::load(&config)
144            .await
145            .unwrap_or_else(metadata::MetadataSnapshot::unavailable);
146        self.index_file_with_metadata(path, &metadata_snapshot)
147            .await
148    }
149
150    async fn index_file_with_metadata(
151        &self,
152        path: &Path,
153        metadata_snapshot: &metadata::MetadataSnapshot,
154    ) -> Result<()> {
155        let content = fs::read_to_string(path).await?;
156        let path_id = path.file_stem().and_then(|stem| stem.to_str());
157        if let Some(header) =
158            parser::parse_header(&content).filter(|header| path_id == Some(header.id.as_str()))
159        {
160            let metadata = metadata_snapshot.record(&header.id).ok();
161            let archived = metadata
162                .as_ref()
163                .map(|m| m.relation == parser::Relation::Archived)
164                .unwrap_or(false);
165            let legacy = metadata
166                .as_ref()
167                .map(|m| m.relation == parser::Relation::Legacy)
168                .unwrap_or(false);
169            let relation_target = metadata
170                .as_ref()
171                .map(|m| m.relation_target.clone())
172                .unwrap_or_default();
173            let info = NoteInfo {
174                id: header.id.clone(),
175                title: header.title.clone(),
176                archived,
177                legacy,
178                alt_id: archived.then(|| relation_target.first().cloned()).flatten(),
179                evo_id: legacy.then(|| relation_target.first().cloned()).flatten(),
180                relation_target,
181                aliases: metadata
182                    .as_ref()
183                    .map(|m| m.aliases.clone())
184                    .unwrap_or_default(),
185                keywords: metadata
186                    .as_ref()
187                    .map(|m| m.keywords.clone())
188                    .unwrap_or_default(),
189                abstract_text: metadata.as_ref().and_then(|m| m.abstract_text.clone()),
190                checklist_status: metadata.as_ref().map(|m| m.checklist_status.clone()),
191                path: path.to_path_buf(),
192            };
193            self.notes.insert(header.id.clone(), info);
194        }
195
196        // Update backlinks from this file.
197        // Convert byte offsets to UTF-16 code-unit offsets (required by LSP) here,
198        // while the line text is available.
199        let lines: Vec<&str> = content.lines().collect();
200        let refs = parser::find_all_refs_filtered(&content);
201        for r in refs {
202            let line_text = lines.get(r.line as usize).copied().unwrap_or("");
203            let loc = BacklinkLocation {
204                file: path.to_path_buf(),
205                line: r.line,
206                start_char: parser::byte_to_utf16(line_text, r.start_char as usize),
207                end_char: parser::byte_to_utf16(line_text, r.end_char as usize),
208            };
209            self.backlinks.entry(r.id).or_default().push(loc);
210        }
211        Ok(())
212    }
213
214    fn remove_backlinks_from(&self, path: &Path) {
215        for mut entry in self.backlinks.iter_mut() {
216            entry.value_mut().retain(|loc| loc.file != path);
217        }
218        // Remove empty entries
219        self.backlinks.retain(|_, v| !v.is_empty());
220    }
221}

229    #[tokio::test]
230    async fn index_file_requires_path_id_to_match_note_header() {
231        let root = std::env::temp_dir().join(format!("zk-lsp-index-{}", std::process::id()));
232        let note_dir = root.join("note");
233        let _ = std::fs::remove_dir_all(&root);
234        std::fs::create_dir_all(&note_dir).unwrap();
235        let path = note_dir.join("2603110000.typ");
236        std::fs::write(
237            &path,
238            concat!(
239                "#import \"../include.typ\": *\n",
240                "#let zk-metadata = zk_metadata(\"2603119999\")\n",
241                "#show: zettel.with(metadata: zk-metadata)\n",
242                "\n",
243                "= Wrong <2603119999>\n",
244            ),
245        )
246        .unwrap();
247
248        let config = Arc::new(RwLock::new(WikiConfig::from_root(root.clone())));
249        let index = NoteIndex::new(config);
250        index.update_file(&path).await.unwrap();
251
252        assert!(index.get("2603119999").is_none());
253        assert!(index.get("2603110000").is_none());
254
255        let _ = std::fs::remove_dir_all(&root);
256    }
