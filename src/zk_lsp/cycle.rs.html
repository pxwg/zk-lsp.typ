3/// Cycle detection on the task dependency graph.
4///
5/// `detect_cycles` runs Tarjan SCC; `render_cycle_errors` formats results for CLI output.
6/// This module has no I/O and does no parsing; it consumes a `DependencyGraph`.
7use std::collections::{HashMap, HashSet};
8use std::io::IsTerminal;

14#[derive(Debug)]
15pub struct DependencyCycle {
16    #[allow(dead_code)]
17    pub nodes: Vec<String>, // note IDs forming the cycle
18    pub edges: Vec<CycleEdgeOccurrence>, // positioned occurrences within this cycle
19}

21// ---------------------------------------------------------------------------
22// Tarjan SCC
23// ---------------------------------------------------------------------------
24
25struct TarjanState<'a> {
26    graph: &'a HashMap<String, Vec<String>>,
27    index_counter: usize,
28    stack: Vec<String>,
29    on_stack: HashMap<String, bool>,
30    index: HashMap<String, usize>,
31    lowlink: HashMap<String, usize>,
32    sccs: Vec<Vec<String>>,
33}
34
35impl<'a> TarjanState<'a> {
36    fn strongconnect(&mut self, v: &str) {
37        self.index.insert(v.to_string(), self.index_counter);
38        self.lowlink.insert(v.to_string(), self.index_counter);
39        self.index_counter += 1;
40        self.stack.push(v.to_string());
41        self.on_stack.insert(v.to_string(), true);
42
43        if let Some(neighbors) = self.graph.get(v) {
44            let neighbors: Vec<String> = neighbors.clone();
45            for w in neighbors {
46                if !self.index.contains_key(&w) {
47                    self.strongconnect(&w);
48                    let lv = *self.lowlink.get(v).unwrap();
49                    let lw = *self.lowlink.get(&w).unwrap_or(&usize::MAX);
50                    self.lowlink.insert(v.to_string(), lv.min(lw));
51                } else if *self.on_stack.get(&w).unwrap_or(&false) {
52                    let lv = *self.lowlink.get(v).unwrap();
53                    let iw = *self.index.get(&w).unwrap();
54                    self.lowlink.insert(v.to_string(), lv.min(iw));
55                }
56            }
57        }
58
59        if self.lowlink.get(v) == self.index.get(v) {
60            let mut scc = Vec::new();
61            loop {
62                let w = self.stack.pop().unwrap();
63                self.on_stack.insert(w.clone(), false);
64                scc.push(w.clone());
65                if w == v {
66                    break;
67                }
68            }
69            self.sccs.push(scc);
70        }
71    }
72}
73
74fn tarjan_sccs(graph: &HashMap<String, Vec<String>>, all_nodes: &[String]) -> Vec<Vec<String>> {
75    let mut state = TarjanState {
76        graph,
77        index_counter: 0,
78        stack: Vec::new(),
79        on_stack: HashMap::new(),
80        index: HashMap::new(),
81        lowlink: HashMap::new(),
82        sccs: Vec::new(),
83    };
84    for v in all_nodes {
85        if !state.index.contains_key(v.as_str()) {
86            state.strongconnect(v);
87        }
88    }
89    state.sccs
90}

96/// Detect cycles in `graph` using Tarjan SCC.
97///
98/// A cycle exists when an SCC has size > 1, or size == 1 with a self-edge.
99/// Each cycle is returned with its member node IDs and all `CycleEdgeOccurrence`s
100/// whose both endpoints lie within the same SCC.
101pub fn detect_cycles(graph: &DependencyGraph) -> Vec<DependencyCycle> {
102    let mut all_nodes: Vec<String> = graph.nodes.clone();
103    all_nodes.sort(); // deterministic iteration order
104    let sccs = tarjan_sccs(&graph.adj, &all_nodes);
105
106    let mut cycles = Vec::new();
107    for scc in sccs {
108        let is_self_loop = scc.len() == 1
109            && graph
110                .adj
111                .get(&scc[0])
112                .map_or(false, |ns| ns.contains(&scc[0]));
113        if scc.len() > 1 || is_self_loop {
114            let scc_set: HashSet<&str> = scc.iter().map(String::as_str).collect();
115            let edges: Vec<CycleEdgeOccurrence> = graph
116                .occurrences
117                .iter()
118                .filter(|occ| {
119                    scc_set.contains(occ.from_note_id.as_str())
120                        && scc_set.contains(occ.to_note_id.as_str())
121                })
122                .cloned()
123                .collect();
124            cycles.push(DependencyCycle { nodes: scc, edges });
125        }
126    }
127    cycles
128}

130/// Render Typst-style error messages for all detected cycles (CLI output).
131///
132/// Column numbers are 1-based byte offsets (not UTF-16; use `byte_to_utf16` for LSP).
133/// ANSI colours are emitted only when stderr is a TTY.
134#[allow(dead_code)]
135pub fn render_cycle_errors(cycles: &[DependencyCycle]) -> String {
136    let color = stderr_color_enabled();
137    let mut out = String::new();
138    for cycle in cycles {
139        // "error: cyclic task dependency detected"
140        if color {
141            out.push_str("\x1b[1;31merror\x1b[0m\x1b[1m: cyclic task dependency detected\x1b[0m\n");
142        } else {
143            out.push_str("error: cyclic task dependency detected\n");
144        }
145
146        for edge in &cycle.edges {
147            let col = edge.byte_start + 1; // 1-based byte column
148            let path = edge.file_path.display();
149            let line_1based = edge.line + 1;
150            // Width of the line-number field for padding
151            let lnum_w = format!("{line_1based}").len();
152            let pad = " ".repeat(lnum_w); // spaces matching line-number width
153
154            // Display-column count of the prefix before '@' (accounts for CJK double-width)
155            let prefix_bytes = edge.byte_start as usize;
156            let display_col =
157                display_width(&edge.line_text[..prefix_bytes.min(edge.line_text.len())]);
158            let underline_disp = display_width(
159                &edge.line_text[prefix_bytes..(edge.byte_end as usize).min(edge.line_text.len())],
160            );
161            let underline = "^".repeat(underline_disp.max(1));
162            let pointer_spaces = " ".repeat(display_col);
163
164            // ┌─ path:line:col
165            if color {
166                out.push_str(&format!(
167                    " \x1b[1;34m{pad}┌─\x1b[0m \x1b[36m{path}:{line_1based}:{col}\x1b[0m\n"
168                ));
169                // blank │
170                out.push_str(&format!(" \x1b[1;34m{pad}│\x1b[0m\n"));
171                // line content
172                out.push_str(&format!(
173                    "\x1b[1;34m{line_1based:>lnum_w$} │\x1b[0m {}\n",
174                    edge.line_text
175                ));
176                // underline
177                out.push_str(&format!(
178                    " \x1b[1;34m{pad}│\x1b[0m {pointer_spaces}\x1b[1;31m{underline}\x1b[0m this dependency participates in a cycle\n"
179                ));
180            } else {
181                out.push_str(&format!(" {pad}┌─ {path}:{line_1based}:{col}\n"));
182                out.push_str(&format!(" {pad}│\n"));
183                out.push_str(&format!("{line_1based:>lnum_w$} │ {}\n", edge.line_text));
184                out.push_str(&format!(
185                    " {pad}│ {pointer_spaces}{underline} this dependency participates in a cycle\n"
186                ));
187            }
188        }
189
190        // Cycle chain
191        let mut chain = cycle.nodes.clone();
192        if let Some(first) = chain.first().cloned() {
193            chain.push(first);
194        }
195        if color {
196            out.push_str(&format!(
197                "\n\x1b[1mcycle\x1b[0m:\n  {}\n\n",
198                chain.join(" \x1b[1;33m->\x1b[0m ")
199            ));
200        } else {
201            out.push_str(&format!("\ncycle:\n  {}\n\n", chain.join(" -> ")));
202        }
203    }
204    out
205}

207fn stderr_color_enabled() -> bool {
208    !cfg!(test) && std::io::stderr().is_terminal()
209}

211/// Count terminal display columns for `s` (CJK chars = 2 columns, others = 1).
212#[allow(dead_code)]
213fn display_width(s: &str) -> usize {
214    s.width()
215}

228    fn simple_notes(pairs: &[(&str, &str, &str)]) -> HashMap<String, (PathBuf, String)> {
229        // pairs: (id, content_without_checklist_header, checklist_body)
230        // We just use raw content strings directly
231        pairs
232            .iter()
233            .map(|(id, _title, content)| {
234                (
235                    id.to_string(),
236                    (PathBuf::from(format!("{id}.typ")), content.to_string()),
237                )
238            })
239            .collect()
240    }
241
242    #[test]
243    fn test_detect_simple_cycle() {
244        // A → B → A
245        let notes = simple_notes(&[
246            ("1111111111", "A", "- [ ] @2222222222\n"),
247            ("2222222222", "B", "- [ ] @1111111111\n"),
248        ]);
249        let graph = build_dependency_graph(&notes);
250        let cycles = detect_cycles(&graph);
251        assert_eq!(cycles.len(), 1, "expected 1 cycle");
252        assert_eq!(cycles[0].nodes.len(), 2);
253        assert_eq!(
254            cycles[0].edges.len(),
255            2,
256            "expected 2 edge occurrences in cycle"
257        );
258    }

260    #[test]
261    fn test_detect_self_loop() {
262        // A → A
263        let notes = simple_notes(&[("1111111111", "A", "- [ ] @1111111111\n")]);
264        let graph = build_dependency_graph(&notes);
265        let cycles = detect_cycles(&graph);
266        assert_eq!(cycles.len(), 1, "expected 1 cycle (self-loop)");
267        assert_eq!(cycles[0].nodes.len(), 1);
268        assert_eq!(cycles[0].edges.len(), 1);
269    }
